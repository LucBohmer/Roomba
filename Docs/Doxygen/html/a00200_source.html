137 static int mqtt3_db_message_store_write(struct mosquitto_db *db, FILE *db_fptr)
138 {
139  uint32_t length;
140  dbid_t i64temp;
141  uint32_t i32temp;
142  uint16_t i16temp, slen, tlen;
143  uint8_t i8temp;
144  struct mosquitto_msg_store *stored;
145  bool force_no_retain;
146 
147  assert(db);
148  assert(db_fptr);
149 
150  stored = db->msg_store;
151  while(stored){
152  if(stored->topic && !strncmp(stored->topic, "$SYS", 4)){
153  if(stored->ref_count <= 1 && stored->dest_id_count == 0){
154  /* $SYS messages that are only retained shouldn't be persisted. */
155  stored = stored->next;
156  continue;
157  }
158  /* Don't save $SYS messages as retained otherwise they can give
159  * misleading information when reloaded. They should still be saved
160  * because a disconnected durable client may have them in their
161  * queue. */
162  force_no_retain = true;
163  }else{
164  force_no_retain = false;
165  }
166  if(stored->topic){
167  tlen = strlen(stored->topic);
168  }else{
169  tlen = 0;
170  }
171  length = htonl(sizeof(dbid_t) + 2+strlen(stored->source_id) +
172  sizeof(uint16_t) + sizeof(uint16_t) +
173  2+tlen + sizeof(uint32_t) +
174  stored->payloadlen + sizeof(uint8_t) + sizeof(uint8_t));
175 
176  i16temp = htons(DB_CHUNK_MSG_STORE);
177  write_e(db_fptr, &i16temp, sizeof(uint16_t));
178  write_e(db_fptr, &length, sizeof(uint32_t));
179 
180  i64temp = stored->db_id;
181  write_e(db_fptr, &i64temp, sizeof(dbid_t));
182 
183  slen = strlen(stored->source_id);
184  i16temp = htons(slen);
185  write_e(db_fptr, &i16temp, sizeof(uint16_t));
186  if(slen){
187  write_e(db_fptr, stored->source_id, slen);
188  }
189 
190  i16temp = htons(stored->source_mid);
191  write_e(db_fptr, &i16temp, sizeof(uint16_t));
192 
193  i16temp = htons(stored->mid);
194  write_e(db_fptr, &i16temp, sizeof(uint16_t));
195 
196  i16temp = htons(tlen);
197  write_e(db_fptr, &i16temp, sizeof(uint16_t));
198  if(tlen){
199  write_e(db_fptr, stored->topic, tlen);
200  }
201 
202  i8temp = (uint8_t )stored->qos;
203  write_e(db_fptr, &i8temp, sizeof(uint8_t));
204 
205  if(force_no_retain == false){
206  i8temp = (uint8_t )stored->retain;
207  }else{
208  i8temp = 0;
209  }
210  write_e(db_fptr, &i8temp, sizeof(uint8_t));
211 
212  i32temp = htonl(stored->payloadlen);
213  write_e(db_fptr, &i32temp, sizeof(uint32_t));
214  if(stored->payloadlen){
215  write_e(db_fptr, stored->payload, (unsigned int)stored->payloadlen);
216  }
217  stored = stored->next;
218  }
219 
220  return MOSQ_ERR_SUCCESS;
221 error:
222  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: %s.", strerror(errno));
223  return 1;

352 int mqtt3_db_backup(struct mosquitto_db *db, bool shutdown)
353 {
354  int rc = 0;
355  FILE *db_fptr = NULL;
356  uint32_t db_version_w = htonl(MOSQ_DB_VERSION);
357  uint32_t crc = htonl(0);
358  dbid_t i64temp;
359  uint32_t i32temp;
360  uint16_t i16temp;
361  uint8_t i8temp;
362  char err[256];
363  char *outfile = NULL;
364  int len;
365 
366  if(!db || !db->config || !db->config->persistence_filepath) return MOSQ_ERR_INVAL;
367  _mosquitto_log_printf(NULL, MOSQ_LOG_INFO, "Saving in-memory database to %s.", db->config->persistence_filepath);
368 
369  len = strlen(db->config->persistence_filepath)+5;
370  outfile = _mosquitto_malloc(len+1);
371  if(!outfile){
372  _mosquitto_log_printf(NULL, MOSQ_LOG_INFO, "Error saving in-memory database, out of memory.");
373  return MOSQ_ERR_NOMEM;
374  }
375  snprintf(outfile, len, "%s.new", db->config->persistence_filepath);
376  outfile[len] = '\0';
377 
378 #ifndef WIN32
379 
398  rc = unlink(outfile);
399  if (rc != 0) {
400  if (errno != ENOENT) {
401  _mosquitto_log_printf(NULL, MOSQ_LOG_INFO, "Error saving in-memory database, unable to remove %s.", outfile);
402  goto error;
403  }
404  }
405 #endif
406 
407  db_fptr = _mosquitto_fopen(outfile, "wb", true);
408  if(db_fptr == NULL){
409  _mosquitto_log_printf(NULL, MOSQ_LOG_INFO, "Error saving in-memory database, unable to open %s for writing.", outfile);
410  goto error;
411  }
412 
413  /* Header */
414  write_e(db_fptr, magic, 15);
415  write_e(db_fptr, &crc, sizeof(uint32_t));
416  write_e(db_fptr, &db_version_w, sizeof(uint32_t));
417 
418  /* DB config */
419  i16temp = htons(DB_CHUNK_CFG);
420  write_e(db_fptr, &i16temp, sizeof(uint16_t));
421  /* chunk length */
422  i32temp = htonl(sizeof(dbid_t) + sizeof(uint8_t) + sizeof(uint8_t));
423  write_e(db_fptr, &i32temp, sizeof(uint32_t));
424  /* db written at broker shutdown or not */
425  i8temp = shutdown;
426  write_e(db_fptr, &i8temp, sizeof(uint8_t));
427  i8temp = sizeof(dbid_t);
428  write_e(db_fptr, &i8temp, sizeof(uint8_t));
429  /* last db mid */
430  i64temp = db->last_db_id;
431  write_e(db_fptr, &i64temp, sizeof(dbid_t));
432 
433  if(mqtt3_db_message_store_write(db, db_fptr)){
434  goto error;
435  }
436 
437  mqtt3_db_client_write(db, db_fptr);
438  mqtt3_db_subs_retain_write(db, db_fptr);
439 
440 #ifndef WIN32
441 
463  fflush(db_fptr);
464  fsync(fileno(db_fptr));
465 #endif
466  fclose(db_fptr);
467 
468 #ifdef WIN32
469  if(remove(db->config->persistence_filepath) != 0){
470  if(errno != ENOENT){
471  goto error;
472  }
473  }
474 #endif
475  if(rename(outfile, db->config->persistence_filepath) != 0){
476  goto error;
477  }
478  _mosquitto_free(outfile);
479  outfile = NULL;
480  return rc;
481 error:
482  if(outfile) _mosquitto_free(outfile);
483  strerror_r(errno, err, 256);
484  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: %s.", err);
485  if(db_fptr) fclose(db_fptr);
486  return 1;
487 }

804 int mqtt3_db_restore(struct mosquitto_db *db)
805 {
806  FILE *fptr;
807  char header[15];
808  int rc = 0;
809  uint32_t crc;
810  dbid_t i64temp;
811  uint32_t i32temp, length;
812  uint16_t i16temp, chunk;
813  uint8_t i8temp;
814  ssize_t rlen;
815  char err[256];
816  struct mosquitto_msg_store_load *load, *load_tmp;
817 
818  assert(db);
819  assert(db->config);
820  assert(db->config->persistence_filepath);
821 
822  db->msg_store_load = NULL;
823 
824  fptr = _mosquitto_fopen(db->config->persistence_filepath, "rb", false);
825  if(fptr == NULL) return MOSQ_ERR_SUCCESS;
826  rlen = fread(&header, 1, 15, fptr);
827  if(rlen == 0){
828  fclose(fptr);
829  _mosquitto_log_printf(NULL, MOSQ_LOG_WARNING, "Warning: Persistence file is empty.");
830  return 0;
831  }else if(rlen != 15){
832  goto error;
833  }
834  if(!memcmp(header, magic, 15)){
835  // Restore DB as normal
836  read_e(fptr, &crc, sizeof(uint32_t));
837  read_e(fptr, &i32temp, sizeof(uint32_t));
838  db_version = ntohl(i32temp);
839  /* IMPORTANT - this is where compatibility checks are made.
840  * Is your DB change still compatible with previous versions?
841  */
842  if(db_version > MOSQ_DB_VERSION && db_version != 0){
843  if(db_version == 2){
844  /* Addition of disconnect_t to client chunk in v3. */
845  }else{
846  fclose(fptr);
847  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unsupported persistent database format version %d (need version %d).", db_version, MOSQ_DB_VERSION);
848  return 1;
849  }
850  }
851 
852  while(rlen = fread(&i16temp, sizeof(uint16_t), 1, fptr), rlen == 1){
853  chunk = ntohs(i16temp);
854  read_e(fptr, &i32temp, sizeof(uint32_t));
855  length = ntohl(i32temp);
856  switch(chunk){
857  case DB_CHUNK_CFG:
858  read_e(fptr, &i8temp, sizeof(uint8_t)); // shutdown
859  read_e(fptr, &i8temp, sizeof(uint8_t)); // sizeof(dbid_t)
860  if(i8temp != sizeof(dbid_t)){
861  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Incompatible database configuration (dbid size is %d bytes, expected %lu)",
862  i8temp, (unsigned long)sizeof(dbid_t));
863  fclose(fptr);
864  return 1;
865  }
866  read_e(fptr, &i64temp, sizeof(dbid_t));
867  db->last_db_id = i64temp;
868  break;
869 
870  case DB_CHUNK_MSG_STORE:
871  if(_db_msg_store_chunk_restore(db, fptr)) return 1;
872  break;
873 
874  case DB_CHUNK_CLIENT_MSG:
875  if(_db_client_msg_chunk_restore(db, fptr)) return 1;
876  break;
877 
878  case DB_CHUNK_RETAIN:
879  if(_db_retain_chunk_restore(db, fptr)) return 1;
880  break;
881 
882  case DB_CHUNK_SUB:
883  if(_db_sub_chunk_restore(db, fptr)) return 1;
884  break;
885 
886  case DB_CHUNK_CLIENT:
887  if(_db_client_chunk_restore(db, fptr)) return 1;
888  break;
889 
890  default:
891  _mosquitto_log_printf(NULL, MOSQ_LOG_WARNING, "Warning: Unsupported chunk \"%d\" in persistent database file. Ignoring.", chunk);
892  fseek(fptr, length, SEEK_CUR);
893  break;
894  }
895  }
896  if(rlen < 0) goto error;
897  }else{
898  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to restore persistent database. Unrecognised file format.");
899  rc = 1;
900  }
901 
902  fclose(fptr);
903 
904  HASH_ITER(hh, db->msg_store_load, load, load_tmp){
905  HASH_DELETE(hh, db->msg_store_load, load);
906  _mosquitto_free(load);
907  }
908  return rc;
909 error:
910  strerror_r(errno, err, 256);
911  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: %s.", err);
912  if(fptr) fclose(fptr);
913  return 1;
914 }
