158 static int callback_mqtt(
159 #else
160 static int callback_mqtt(struct libwebsocket_context *context,
161 #endif
162  struct libwebsocket *wsi,
163  enum libwebsocket_callback_reasons reason,
164  void *user,
165  void *in,
166  size_t len)
167 {
168  struct mosquitto_db *db;
169  struct mosquitto *mosq = NULL;
170  struct _mosquitto_packet *packet;
171  int count, i, j;
172  const struct libwebsocket_protocols *p;
173  struct libws_mqtt_data *u = (struct libws_mqtt_data *)user;
174  size_t pos;
175  uint8_t *buf;
176  int rc;
177  uint8_t byte;
178 
179  db = &int_db;
180 
181  switch (reason) {
182  case LWS_CALLBACK_ESTABLISHED:
183  mosq = mqtt3_context_init(db, WEBSOCKET_CLIENT);
184  if(mosq){
185  p = libwebsockets_get_protocol(wsi);
186  for (i=0; i<db->config->listener_count; i++){
187  if (db->config->listeners[i].protocol == mp_websockets) {
188  for (j=0; db->config->listeners[i].ws_protocol[j].name; j++){
189  if (p == &db->config->listeners[i].ws_protocol[j]){
190  mosq->listener = &db->config->listeners[i];
191  mosq->listener->client_count++;
192  }
193  }
194  }
195  }
196  if(!mosq->listener){
197  _mosquitto_free(mosq);
198  return -1;
199  }
200 #if !defined(LWS_LIBRARY_VERSION_NUMBER)
201  mosq->ws_context = context;
202 #endif
203  mosq->wsi = wsi;
204  u->mosq = mosq;
205  }else{
206  return -1;
207  }
208  easy_address(libwebsocket_get_socket_fd(wsi), mosq);
209  if(!mosq->address){
210  /* getpeername and inet_ntop failed and not a bridge */
211  _mosquitto_free(mosq);
212  u->mosq = NULL;
213  return -1;
214  }
215  if(mosq->listener->max_connections > 0 && mosq->listener->client_count > mosq->listener->max_connections){
216  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "Client connection from %s denied: max_connections exceeded.", mosq->address);
217  _mosquitto_free(mosq);
218  u->mosq = NULL;
219  return -1;
220  }
221  mosq->sock = libwebsocket_get_socket_fd(wsi);
222  HASH_ADD(hh_sock, db->contexts_by_sock, sock, sizeof(mosq->sock), mosq);
223  break;
224 
225  case LWS_CALLBACK_CLOSED:
226  if(!u){
227  return -1;
228  }
229  mosq = u->mosq;
230  if(mosq){
231  if(mosq->sock > 0){
232  HASH_DELETE(hh_sock, db->contexts_by_sock, mosq);
233  mosq->sock = INVALID_SOCKET;
234  mosq->pollfd_index = -1;
235  }
236  mosq->wsi = NULL;
237  do_disconnect(db, mosq);
238  }
239  break;
240 
241  case LWS_CALLBACK_SERVER_WRITEABLE:
242  if(!u){
243  return -1;
244  }
245  mosq = u->mosq;
246  if(!mosq || mosq->state == mosq_cs_disconnect_ws || mosq->state == mosq_cs_disconnecting){
247  return -1;
248  }
249 
250  mqtt3_db_message_write(db, mosq);
251 
252  if(mosq->out_packet && !mosq->current_out_packet){
253  mosq->current_out_packet = mosq->out_packet;
254  mosq->out_packet = mosq->out_packet->next;
255  if(!mosq->out_packet){
256  mosq->out_packet_last = NULL;
257  }
258  }
259 
260  if(mosq->current_out_packet && !lws_send_pipe_choked(mosq->wsi)){
261  packet = mosq->current_out_packet;
262 
263  if(packet->pos == 0 && packet->to_process == packet->packet_length){
264  /* First time this packet has been dealt with.
265  * libwebsockets requires that the payload has
266  * LWS_SEND_BUFFER_PRE_PADDING space available before the
267  * actual data and LWS_SEND_BUFFER_POST_PADDING afterwards.
268  * We've already made the payload big enough to allow this,
269  * but need to move it into position here. */
270  memmove(&packet->payload[LWS_SEND_BUFFER_PRE_PADDING], packet->payload, packet->packet_length);
271  packet->pos += LWS_SEND_BUFFER_PRE_PADDING;
272  }
273  count = libwebsocket_write(wsi, &packet->payload[packet->pos], packet->to_process, LWS_WRITE_BINARY);
274 #ifdef LWS_IS_OLD
275  /* lws < 1.3 doesn't return a valid count, assume everything sent. */
276  count = packet->to_process;
277 #endif
278  if(count < 0){
279  return 0;
280  }
281  packet->to_process -= count;
282  packet->pos += count;
283  if(packet->to_process > 0){
284  break;
285  }
286 
287  /* Free data and reset values */
288  mosq->current_out_packet = mosq->out_packet;
289  if(mosq->out_packet){
290  mosq->out_packet = mosq->out_packet->next;
291  if(!mosq->out_packet){
292  mosq->out_packet_last = NULL;
293  }
294  }
295 
296  _mosquitto_packet_cleanup(packet);
297  _mosquitto_free(packet);
298 
299  mosq->next_msg_out = mosquitto_time() + mosq->keepalive;
300  }
301  if(mosq->current_out_packet){
302  libwebsocket_callback_on_writable(mosq->ws_context, mosq->wsi);
303  }
304  break;
305 
306  case LWS_CALLBACK_RECEIVE:
307  if(!u || !u->mosq){
308  return -1;
309  }
310  mosq = u->mosq;
311  pos = 0;
312  buf = (uint8_t *)in;
313 #ifdef WITH_SYS_TREE
314  g_bytes_received += len;
315 #endif
316  while(pos < len){
317  if(!mosq->in_packet.command){
318  mosq->in_packet.command = buf[pos];
319  pos++;
320  /* Clients must send CONNECT as their first command. */
321  if(mosq->state == mosq_cs_new && (mosq->in_packet.command&0xF0) != CONNECT){
322  return -1;
323  }
324  }
325  if(mosq->in_packet.remaining_count <= 0){
326  do{
327  if(pos == len){
328  return 0;
329  }
330  byte = buf[pos];
331  pos++;
332 
333  mosq->in_packet.remaining_count--;
334  /* Max 4 bytes length for remaining length as defined by protocol.
335  * Anything more likely means a broken/malicious client.
336  */
337  if(mosq->in_packet.remaining_count < -4){
338  return -1;
339  }
340 
341  mosq->in_packet.remaining_length += (byte & 127) * mosq->in_packet.remaining_mult;
342  mosq->in_packet.remaining_mult *= 128;
343  }while((byte & 128) != 0);
344  mosq->in_packet.remaining_count *= -1;
345 
346  if(mosq->in_packet.remaining_length > 0){
347  mosq->in_packet.payload = _mosquitto_malloc(mosq->in_packet.remaining_length*sizeof(uint8_t));
348  if(!mosq->in_packet.payload){
349  return -1;
350  }
351  mosq->in_packet.to_process = mosq->in_packet.remaining_length;
352  }
353  }
354  if(mosq->in_packet.to_process>0){
355  if(len - pos >= mosq->in_packet.to_process){
356  memcpy(&mosq->in_packet.payload[mosq->in_packet.pos], &buf[pos], mosq->in_packet.to_process);
357  mosq->in_packet.pos += mosq->in_packet.to_process;
358  pos += mosq->in_packet.to_process;
359  mosq->in_packet.to_process = 0;
360  }else{
361  memcpy(&mosq->in_packet.payload[mosq->in_packet.pos], &buf[pos], len-pos);
362  mosq->in_packet.pos += len-pos;
363  mosq->in_packet.to_process -= len-pos;
364  return 0;
365  }
366  }
367  /* All data for this packet is read. */
368  mosq->in_packet.pos = 0;
369 #ifdef WITH_SYS_TREE
370  g_msgs_received++;
371  if(((mosq->in_packet.command)&0xF5) == PUBLISH){
372  g_pub_msgs_received++;
373  }
374 #endif
375  rc = mqtt3_packet_handle(db, mosq);
376 
377  /* Free data and reset values */
378  _mosquitto_packet_cleanup(&mosq->in_packet);
379 
380  mosq->last_msg_in = mosquitto_time();
381 
382  if(rc){
383  do_disconnect(db, mosq);
384  return -1;
385  }
386  }
387  break;
388 
389  default:
390  break;
391  }
392 
393  return 0;
394 }

493 #endif

502

528  "Server: mosquitto\r\n"
529  "Content-Length: %u\r\n\r\n",

532  fclose(u->fptr);
533  u->fptr = NULL;
534  return -1;
535  }

537  break;
538

541  /* For extra POST data? */
542  return -1;

545  /* For end of extra POST data? */

578  return -1;
579  }
580  break;

582  case LWS_CALLBACK_CLOSED:
583  case LWS_CALLBACK_CLOSED_HTTP:
