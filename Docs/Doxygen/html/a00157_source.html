836 int mosquitto_loop(struct mosquitto *mosq, int timeout, int max_packets)
837 {
838 #ifdef HAVE_PSELECT
839  struct timespec local_timeout;
840 #else
841  struct timeval local_timeout;
842 #endif
843  fd_set readfds, writefds;
844  int fdcount;
845  int rc;
846  char pairbuf;
847  int maxfd = 0;
848  time_t now;
849 
850  if(!mosq || max_packets < 1) return MOSQ_ERR_INVAL;
851 #ifndef WIN32
852  if(mosq->sock >= FD_SETSIZE || mosq->sockpairR >= FD_SETSIZE){
853  return MOSQ_ERR_INVAL;
854  }
855 #endif
856 
857  FD_ZERO(&readfds);
858  FD_ZERO(&writefds);
859  if(mosq->sock != INVALID_SOCKET){
860  maxfd = mosq->sock;
861  FD_SET(mosq->sock, &readfds);
862  pthread_mutex_lock(&mosq->current_out_packet_mutex);
863  pthread_mutex_lock(&mosq->out_packet_mutex);
864  if(mosq->out_packet || mosq->current_out_packet){
865  FD_SET(mosq->sock, &writefds);
866  }
867 #ifdef WITH_TLS
868  if(mosq->ssl){
869  if(mosq->want_write){
870  FD_SET(mosq->sock, &writefds);
871  }else if(mosq->want_connect){
872  /* Remove possible FD_SET from above, we don't want to check
873  * for writing if we are still connecting, unless want_write is
874  * definitely set. The presence of outgoing packets does not
875  * matter yet. */
876  FD_CLR(mosq->sock, &writefds);
877  }
878  }
879 #endif
880  pthread_mutex_unlock(&mosq->out_packet_mutex);
881  pthread_mutex_unlock(&mosq->current_out_packet_mutex);
882  }else{
883 #ifdef WITH_SRV
884  if(mosq->achan){
885  pthread_mutex_lock(&mosq->state_mutex);
886  if(mosq->state == mosq_cs_connect_srv){
887  rc = ares_fds(mosq->achan, &readfds, &writefds);
888  if(rc > maxfd){
889  maxfd = rc;
890  }
891  }else{
892  pthread_mutex_unlock(&mosq->state_mutex);
893  return MOSQ_ERR_NO_CONN;
894  }
895  pthread_mutex_unlock(&mosq->state_mutex);
896  }
897 #else
898  return MOSQ_ERR_NO_CONN;
899 #endif
900  }
901  if(mosq->sockpairR != INVALID_SOCKET){
902  /* sockpairR is used to break out of select() before the timeout, on a
903  * call to publish() etc. */
904  FD_SET(mosq->sockpairR, &readfds);
905  if(mosq->sockpairR > maxfd){
906  maxfd = mosq->sockpairR;
907  }
908  }
909 
910  if(timeout < 0){
911  timeout = 1000;
912  }
913 
914  now = mosquitto_time();
915  if(mosq->next_msg_out && now + timeout/1000 > mosq->next_msg_out){
916  timeout = (mosq->next_msg_out - now)*1000;
917  }
918 
919  if(timeout < 0){
920  /* There has been a delay somewhere which means we should have already
921  * sent a message. */
922  timeout = 0;
923  }
924 
925  local_timeout.tv_sec = timeout/1000;
926 #ifdef HAVE_PSELECT
927  local_timeout.tv_nsec = (timeout-local_timeout.tv_sec*1000)*1e6;
928 #else
929  local_timeout.tv_usec = (timeout-local_timeout.tv_sec*1000)*1000;
930 #endif
931 
932 #ifdef HAVE_PSELECT
933  fdcount = pselect(maxfd+1, &readfds, &writefds, NULL, &local_timeout, NULL);
934 #else
935  fdcount = select(maxfd+1, &readfds, &writefds, NULL, &local_timeout);
936 #endif
937  if(fdcount == -1){
938 #ifdef WIN32
939  errno = WSAGetLastError();
940 #endif
941  if(errno == EINTR){
942  return MOSQ_ERR_SUCCESS;
943  }else{
944  return MOSQ_ERR_ERRNO;
945  }
946  }else{
947  if(mosq->sock != INVALID_SOCKET){
948  if(FD_ISSET(mosq->sock, &readfds)){
949 #ifdef WITH_TLS
950  if(mosq->want_connect){
951  rc = mosquitto__socket_connect_tls(mosq);
952  if(rc) return rc;
953  }else
954 #endif
955  {
956  do{
957  rc = mosquitto_loop_read(mosq, max_packets);
958  if(rc || mosq->sock == INVALID_SOCKET){
959  return rc;
960  }
961  }while(SSL_DATA_PENDING(mosq));
962  }
963  }
964  if(mosq->sockpairR != INVALID_SOCKET && FD_ISSET(mosq->sockpairR, &readfds)){
965 #ifndef WIN32
966  if(read(mosq->sockpairR, &pairbuf, 1) == 0){
967  }
968 #else
969  recv(mosq->sockpairR, &pairbuf, 1, 0);
970 #endif
971  /* Fake write possible, to stimulate output write even though
972  * we didn't ask for it, because at that point the publish or
973  * other command wasn't present. */
974  FD_SET(mosq->sock, &writefds);
975  }
976  if(FD_ISSET(mosq->sock, &writefds)){
977 #ifdef WITH_TLS
978  if(mosq->want_connect){
979  rc = mosquitto__socket_connect_tls(mosq);
980  if(rc) return rc;
981  }else
982 #endif
983  {
984  rc = mosquitto_loop_write(mosq, max_packets);
985  if(rc || mosq->sock == INVALID_SOCKET){
986  return rc;
987  }
988  }
989  }
990  }
991 #ifdef WITH_SRV
992  if(mosq->achan){
993  ares_process(mosq->achan, &readfds, &writefds);
994  }
995 #endif
996  }
997  return mosquitto_loop_misc(mosq);
998 }

1000 int mosquitto_loop_forever(struct mosquitto *mosq, int timeout, int max_packets)
1001 {
1002  int run = 1;
1003  int rc;
1004  unsigned int reconnects = 0;
1005  unsigned long reconnect_delay;
1006 
1007  if(!mosq) return MOSQ_ERR_INVAL;
1008 
1009  if(mosq->state == mosq_cs_connect_async){
1010  mosquitto_reconnect(mosq);
1011  }
1012 
1013  while(run){
1014  do{
1015  rc = mosquitto_loop(mosq, timeout, max_packets);
1016  if (reconnects !=0 && rc == MOSQ_ERR_SUCCESS){
1017  reconnects = 0;
1018  }
1019  }while(run && rc == MOSQ_ERR_SUCCESS);
1020  /* Quit after fatal errors. */
1021  switch(rc){
1022  case MOSQ_ERR_NOMEM:
1023  case MOSQ_ERR_PROTOCOL:
1024  case MOSQ_ERR_INVAL:
1025  case MOSQ_ERR_NOT_FOUND:
1026  case MOSQ_ERR_TLS:
1027  case MOSQ_ERR_PAYLOAD_SIZE:
1028  case MOSQ_ERR_NOT_SUPPORTED:
1029  case MOSQ_ERR_AUTH:
1030  case MOSQ_ERR_ACL_DENIED:
1031  case MOSQ_ERR_UNKNOWN:
1032  case MOSQ_ERR_EAI:
1033  case MOSQ_ERR_PROXY:
1034  return rc;
1035  case MOSQ_ERR_ERRNO:
1036  break;
1037  }
1038  if(errno == EPROTO){
1039  return rc;
1040  }
1041  do{
1042  rc = MOSQ_ERR_SUCCESS;
1043  pthread_mutex_lock(&mosq->state_mutex);
1044  if(mosq->state == mosq_cs_disconnecting){
1045  run = 0;
1046  pthread_mutex_unlock(&mosq->state_mutex);
1047  }else{
1048  pthread_mutex_unlock(&mosq->state_mutex);
1049 
1050  if(mosq->reconnect_delay > 0 && mosq->reconnect_exponential_backoff){
1051  reconnect_delay = mosq->reconnect_delay*reconnects*reconnects;
1052  }else{
1053  reconnect_delay = mosq->reconnect_delay;
1054  }
1055 
1056  if(reconnect_delay > mosq->reconnect_delay_max){
1057  reconnect_delay = mosq->reconnect_delay_max;
1058  }else{
1059  reconnects++;
1060  }
1061 
1062 #ifdef WIN32
1063  Sleep(reconnect_delay*1000);
1064 #else
1065  sleep(reconnect_delay);
1066 #endif
1067 
1068  pthread_mutex_lock(&mosq->state_mutex);
1069  if(mosq->state == mosq_cs_disconnecting){
1070  run = 0;
1071  pthread_mutex_unlock(&mosq->state_mutex);
1072  }else{
1073  pthread_mutex_unlock(&mosq->state_mutex);
1074  rc = mosquitto_reconnect(mosq);
1075  }
1076  }
1077  }while(run && rc != MOSQ_ERR_SUCCESS);
1078  }
1079  return rc;
1080 }
