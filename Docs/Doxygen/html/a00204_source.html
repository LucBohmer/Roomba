224 int mosquitto_acl_check_default(struct mosquitto_db *db, struct mosquitto *context, const char *topic, int access)
225 {
226  char *local_acl;
227  struct _mosquitto_acl *acl_root;
228  bool result;
229  int i;
230  int len, tlen, clen, ulen;
231  char *s;
232 
233  if(!db || !context || !topic) return MOSQ_ERR_INVAL;
234  if(!db->acl_list && !db->acl_patterns) return MOSQ_ERR_SUCCESS;
235  if(context->bridge) return MOSQ_ERR_SUCCESS;
236  if(!context->acl_list && !db->acl_patterns) return MOSQ_ERR_ACL_DENIED;
237 
238  if(context->acl_list){
239  acl_root = context->acl_list->acl;
240  }else{
241  acl_root = NULL;
242  }
243 
244  /* Loop through all ACLs for this client. */
245  while(acl_root){
246  /* Loop through the topic looking for matches to this ACL. */
247 
248  /* If subscription starts with $, acl_root->topic must also start with $. */
249  if(topic[0] == '$' && acl_root->topic[0] != '$'){
250  acl_root = acl_root->next;
251  continue;
252  }
253  mosquitto_topic_matches_sub(acl_root->topic, topic, &result);
254  if(result){
255  if(access & acl_root->access){
256  /* And access is allowed. */
257  return MOSQ_ERR_SUCCESS;
258  }
259  }
260  acl_root = acl_root->next;
261  }
262 
263  acl_root = db->acl_patterns;
264 
265  if(acl_root){
266  /* We are using pattern based acls. Check whether the username or
267  * client id contains a + or # and if so deny access.
268  *
269  * Without this, a malicious client may configure its username/client
270  * id to bypass ACL checks (or have a username/client id that cannot
271  * publish or receive messages to its own place in the hierarchy).
272  */
273  if(context->username && strpbrk(context->username, "+#")){
274  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "ACL denying access to client with dangerous username \"%s\"", context->username);
275  return MOSQ_ERR_ACL_DENIED;
276  }
277 
278  if(context->id && strpbrk(context->id, "+#")){
279  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "ACL denying access to client with dangerous client id \"%s\"", context->id);
280  return MOSQ_ERR_ACL_DENIED;
281  }
282  }
283 
284  /* Loop through all pattern ACLs. */
285  clen = strlen(context->id);
286  while(acl_root){
287  tlen = strlen(acl_root->topic);
288 
289  if(acl_root->ucount && !context->username){
290  acl_root = acl_root->next;
291  continue;
292  }
293 
294  if(context->username){
295  ulen = strlen(context->username);
296  len = tlen + acl_root->ccount*(clen-2) + acl_root->ucount*(ulen-2);
297  }else{
298  ulen = 0;
299  len = tlen + acl_root->ccount*(clen-2);
300  }
301  local_acl = _mosquitto_malloc(len+1);
302  if(!local_acl) return 1; // FIXME
303  s = local_acl;
304  for(i=0; i<tlen; i++){
305  if(i<tlen-1 && acl_root->topic[i] == '%'){
306  if(acl_root->topic[i+1] == 'c'){
307  i++;
308  strncpy(s, context->id, clen);
309  s+=clen;
310  continue;
311  }else if(context->username && acl_root->topic[i+1] == 'u'){
312  i++;
313  strncpy(s, context->username, ulen);
314  s+=ulen;
315  continue;
316  }
317  }
318  s[0] = acl_root->topic[i];
319  s++;
320  }
321  local_acl[len] = '\0';
322 
323  mosquitto_topic_matches_sub(local_acl, topic, &result);
324  _mosquitto_free(local_acl);
325  if(result){
326  if(access & acl_root->access){
327  /* And access is allowed. */
328  return MOSQ_ERR_SUCCESS;
329  }
330  }
331 
332  acl_root = acl_root->next;
333  }
334 
335  return MOSQ_ERR_ACL_DENIED;
336 }

338 static int _aclfile_parse(struct mosquitto_db *db)
339 {
340  FILE *aclfile;
341  char buf[1024];
342  char *token;
343  char *user = NULL;
344  char *topic;
345  char *access_s;
346  int access;
347  int rc;
348  int slen;
349  int topic_pattern;
350  char *saveptr = NULL;
351 
352  if(!db || !db->config) return MOSQ_ERR_INVAL;
353  if(!db->config->acl_file) return MOSQ_ERR_SUCCESS;
354 
355  aclfile = _mosquitto_fopen(db->config->acl_file, "rt", false);
356  if(!aclfile){
357  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to open acl_file \"%s\".", db->config->acl_file);
358  return 1;
359  }
360 
361  // topic [read|write] <topic>
362  // user <user>
363 
364  while(fgets(buf, 1024, aclfile)){
365  slen = strlen(buf);
366  while(slen > 0 && (buf[slen-1] == 10 || buf[slen-1] == 13)){
367  buf[slen-1] = '\0';
368  slen = strlen(buf);
369  }
370  if(buf[0] == '#'){
371  continue;
372  }
373  token = strtok_r(buf, " ", &saveptr);
374  if(token){
375  if(!strcmp(token, "topic") || !strcmp(token, "pattern")){
376  if(!strcmp(token, "topic")){
377  topic_pattern = 0;
378  }else{
379  topic_pattern = 1;
380  }
381 
382  access_s = strtok_r(NULL, " ", &saveptr);
383  if(!access_s){
384  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Empty topic in acl_file.");
385  if(user) _mosquitto_free(user);
386  fclose(aclfile);
387  return MOSQ_ERR_INVAL;
388  }
389  token = strtok_r(NULL, "", &saveptr);
390  if(token){
391  topic = token;
392  /* Ignore duplicate spaces */
393  while(topic[0] == ' '){
394  topic++;
395  }
396  }else{
397  topic = access_s;
398  access_s = NULL;
399  }
400  if(access_s){
401  if(!strcmp(access_s, "read")){
402  access = MOSQ_ACL_READ;
403  }else if(!strcmp(access_s, "write")){
404  access = MOSQ_ACL_WRITE;
405  }else if(!strcmp(access_s, "readwrite")){
406  access = MOSQ_ACL_READ | MOSQ_ACL_WRITE;
407  }else{
408  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Invalid topic access type \"%s\" in acl_file.", access_s);
409  if(user) _mosquitto_free(user);
410  fclose(aclfile);
411  return MOSQ_ERR_INVAL;
412  }
413  }else{
414  access = MOSQ_ACL_READ | MOSQ_ACL_WRITE;
415  }
416  if(topic_pattern == 0){
417  rc = _add_acl(db, user, topic, access);
418  }else{
419  rc = _add_acl_pattern(db, topic, access);
420  }
421  if(rc){
422  if(user) _mosquitto_free(user);
423  fclose(aclfile);
424  return rc;
425  }
426  }else if(!strcmp(token, "user")){
427  token = strtok_r(NULL, "", &saveptr);
428  if(token){
429  /* Ignore duplicate spaces */
430  while(token[0] == ' '){
431  token++;
432  }
433  if(user) _mosquitto_free(user);
434  user = _mosquitto_strdup(token);
435  if(!user){
436  fclose(aclfile);
437  return MOSQ_ERR_NOMEM;
438  }
439  }else{
440  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Missing username in acl_file.");
441  if(user) _mosquitto_free(user);
442  fclose(aclfile);
443  return 1;
444  }
445  }
446  }
447  }
448 
449  if(user) _mosquitto_free(user);
450  fclose(aclfile);
451 
452  return MOSQ_ERR_SUCCESS;
453 }

468 static int _acl_cleanup(struct mosquitto_db *db, bool reload)
469 {
470  struct mosquitto *context, *ctxt_tmp;
471  struct _mosquitto_acl_user *user_tail;
472 
473  if(!db) return MOSQ_ERR_INVAL;
474  if(!db->acl_list) return MOSQ_ERR_SUCCESS;
475 
476  /* As we're freeing ACLs, we must clear context->acl_list to ensure no
477  * invalid memory accesses take place later.
478  * This *requires* the ACLs to be reapplied after _acl_cleanup()
479  * is called if we are reloading the config. If this is not done, all
480  * access will be denied to currently connected clients.
481  */
482  HASH_ITER(hh_id, db->contexts_by_id, context, ctxt_tmp){
483  context->acl_list = NULL;
484  }
485 
486  while(db->acl_list){
487  user_tail = db->acl_list->next;
488 
489  _free_acl(db->acl_list->acl);
490  if(db->acl_list->username){
491  _mosquitto_free(db->acl_list->username);
492  }
493  _mosquitto_free(db->acl_list);
494 
495  db->acl_list = user_tail;
496  }
497 
498  if(db->acl_patterns){
499  _free_acl(db->acl_patterns);
500  db->acl_patterns = NULL;
501  }
502  return MOSQ_ERR_SUCCESS;
503 }

505 static int _pwfile_parse(const char *file, struct _mosquitto_unpwd **root)
506 {
507  FILE *pwfile;
508  struct _mosquitto_unpwd *unpwd;
509  char buf[256];
510  char *username, *password;
511  int len;
512  char *saveptr = NULL;
513 
514  pwfile = _mosquitto_fopen(file, "rt", false);
515  if(!pwfile){
516  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to open pwfile \"%s\".", file);
517  return 1;
518  }
519 
520  while(!feof(pwfile)){
521  if(fgets(buf, 256, pwfile)){
522  username = strtok_r(buf, ":", &saveptr);
523  if(username){
524  unpwd = _mosquitto_calloc(1, sizeof(struct _mosquitto_unpwd));
525  if(!unpwd){
526  fclose(pwfile);
527  return MOSQ_ERR_NOMEM;
528  }
529  unpwd->username = _mosquitto_strdup(username);
530  if(!unpwd->username){
531  _mosquitto_free(unpwd);
532  fclose(pwfile);
533  return MOSQ_ERR_NOMEM;
534  }
535  len = strlen(unpwd->username);
536  while(unpwd->username[len-1] == 10 || unpwd->username[len-1] == 13){
537  unpwd->username[len-1] = '\0';
538  len = strlen(unpwd->username);
539  }
540  password = strtok_r(NULL, ":", &saveptr);
541  if(password){
542  unpwd->password = _mosquitto_strdup(password);
543  if(!unpwd->password){
544  fclose(pwfile);
545  _mosquitto_free(unpwd->username);
546  _mosquitto_free(unpwd);
547  return MOSQ_ERR_NOMEM;
548  }
549  len = strlen(unpwd->password);
550  while(len && (unpwd->password[len-1] == 10 || unpwd->password[len-1] == 13)){
551  unpwd->password[len-1] = '\0';
552  len = strlen(unpwd->password);
553  }
554  }
555  HASH_ADD_KEYPTR(hh, *root, unpwd->username, strlen(unpwd->username), unpwd);
556  }
557  }
558  }
559  fclose(pwfile);
560 
561  return MOSQ_ERR_SUCCESS;
562 }

610  return MOSQ_ERR_INVAL;
611  }

614  return MOSQ_ERR_INVAL;

618  return MOSQ_ERR_INVAL;

622 #endif

653 int mosquitto_unpwd_check_default(struct mosquitto_db *db, const char *username, const char *password)
654 {
655  struct _mosquitto_unpwd *u, *tmp;
656 #ifdef WITH_TLS
657  unsigned char hash[EVP_MAX_MD_SIZE];
658  unsigned int hash_len;
659  int rc;
660 #endif
661 
662  if(!db) return MOSQ_ERR_INVAL;
663  if(!db->unpwd) return MOSQ_ERR_SUCCESS;
664  if(!username) return MOSQ_ERR_INVAL; /* Check must be made only after checking db->unpwd. */
665 
666  HASH_ITER(hh, db->unpwd, u, tmp){
667  if(!strcmp(u->username, username)){
668  if(u->password){

720  * Includes:
721  * - Disconnecting anonymous users if appropriate
722  * - Disconnecting users with invalid passwords
723  * - Reapplying ACLs
724  */
725 int mosquitto_security_apply_default(struct mosquitto_db *db)
726 {
727  struct mosquitto *context, *ctxt_tmp;
728  struct _mosquitto_acl_user *acl_user_tail;
729  bool allow_anonymous;
730 
731  if(!db) return MOSQ_ERR_INVAL;
732 
733  allow_anonymous = db->config->allow_anonymous;
734 
735  HASH_ITER(hh_id, db->contexts_by_id, context, ctxt_tmp){
736  /* Check for anonymous clients when allow_anonymous is false */
737  if(!allow_anonymous && !context->username){
738  context->state = mosq_cs_disconnecting;
739  do_disconnect(db, context);
740  continue;
741  }
742  /* Check for connected clients that are no longer authorised */
743  if(mosquitto_unpwd_check_default(db, context->username, context->password) != MOSQ_ERR_SUCCESS){
744  context->state = mosq_cs_disconnecting;
745  do_disconnect(db, context);
746  continue;
747  }
748  /* Check for ACLs and apply to user. */
749  if(db->acl_list){
750  acl_user_tail = db->acl_list;
751  while(acl_user_tail){
752  if(acl_user_tail->username){
753  if(context->username){
754  if(!strcmp(acl_user_tail->username, context->username)){
755  context->acl_list = acl_user_tail;
756  break;
757  }
758  }
759  }else{
760  if(!context->username){
761  context->acl_list = acl_user_tail;
762  break;
763  }
764  }
765  acl_user_tail = acl_user_tail->next;
766  }
767  }
768  }
769  return MOSQ_ERR_SUCCESS;
