53 int mqtt3_log_init(struct mqtt3_config *config)
54 {
55  int rc = 0;
56 
57  log_priorities = config->log_type;
58  log_destinations = config->log_dest;
59 
60  if(log_destinations & MQTT3_LOG_SYSLOG){
61 #ifndef WIN32
62  openlog("mosquitto", LOG_PID|LOG_CONS, config->log_facility);
63 #else
64  syslog_h = OpenEventLog(NULL, "mosquitto");
65 #endif
66  }
67 
68  if(log_destinations & MQTT3_LOG_FILE){
69  if(drop_privileges(config, true)){
70  return 1;
71  }
72  config->log_fptr = _mosquitto_fopen(config->log_file, "at", true);
73  if(!config->log_fptr){
74  log_destinations = MQTT3_LOG_STDERR;
75  log_priorities = MOSQ_LOG_ERR;
76  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to open log file %s for writing.", config->log_file);
77  return MOSQ_ERR_INVAL;
78  }
79  restore_privileges();
80  }
81  return rc;
82 }

104 int _mosquitto_log_vprintf(struct mosquitto *mosq, int priority, const char *fmt, va_list va)
105 {
106  char *s;
107  char *st;
108  int len;
109 #ifdef WIN32
110  char *sp;
111 #endif
112  const char *topic;
113  int syslog_priority;
114  time_t now = time(NULL);
115  static time_t last_flush = 0;
116 
117  if((log_priorities & priority) && log_destinations != MQTT3_LOG_NONE){
118  switch(priority){
119  case MOSQ_LOG_SUBSCRIBE:
120  topic = "$SYS/broker/log/M/subscribe";
121 #ifndef WIN32
122  syslog_priority = LOG_NOTICE;
123 #else
124  syslog_priority = EVENTLOG_INFORMATION_TYPE;
125 #endif
126  break;
127  case MOSQ_LOG_UNSUBSCRIBE:
128  topic = "$SYS/broker/log/M/unsubscribe";
129 #ifndef WIN32
130  syslog_priority = LOG_NOTICE;
131 #else
132  syslog_priority = EVENTLOG_INFORMATION_TYPE;
133 #endif
134  break;
135  case MOSQ_LOG_DEBUG:
136  topic = "$SYS/broker/log/D";
137 #ifndef WIN32
138  syslog_priority = LOG_DEBUG;
139 #else
140  syslog_priority = EVENTLOG_INFORMATION_TYPE;
141 #endif
142  break;
143  case MOSQ_LOG_ERR:
144  topic = "$SYS/broker/log/E";
145 #ifndef WIN32
146  syslog_priority = LOG_ERR;
147 #else
148  syslog_priority = EVENTLOG_ERROR_TYPE;
149 #endif
150  break;
151  case MOSQ_LOG_WARNING:
152  topic = "$SYS/broker/log/W";
153 #ifndef WIN32
154  syslog_priority = LOG_WARNING;
155 #else
156  syslog_priority = EVENTLOG_WARNING_TYPE;
157 #endif
158  break;
159  case MOSQ_LOG_NOTICE:
160  topic = "$SYS/broker/log/N";
161 #ifndef WIN32
162  syslog_priority = LOG_NOTICE;
163 #else
164  syslog_priority = EVENTLOG_INFORMATION_TYPE;
165 #endif
166  break;
167  case MOSQ_LOG_INFO:
168  topic = "$SYS/broker/log/I";
169 #ifndef WIN32
170  syslog_priority = LOG_INFO;
171 #else
172  syslog_priority = EVENTLOG_INFORMATION_TYPE;
173 #endif
174  break;
175 #ifdef WITH_WEBSOCKETS
176  case MOSQ_LOG_WEBSOCKETS:
177  topic = "$SYS/broker/log/WS";
178 #ifndef WIN32
179  syslog_priority = LOG_DEBUG;
180 #else
181  syslog_priority = EVENTLOG_INFORMATION_TYPE;
182 #endif
183  break;
184 #endif
185  default:
186  topic = "$SYS/broker/log/E";
187 #ifndef WIN32
188  syslog_priority = LOG_ERR;
189 #else
190  syslog_priority = EVENTLOG_ERROR_TYPE;
191 #endif
192  }
193  len = strlen(fmt) + 500;
194  s = _mosquitto_malloc(len*sizeof(char));
195  if(!s) return MOSQ_ERR_NOMEM;
196 
197  vsnprintf(s, len, fmt, va);
198  s[len-1] = '\0'; /* Ensure string is null terminated. */
199 
200  if(log_destinations & MQTT3_LOG_STDOUT){
201  if(int_db.config && int_db.config->log_timestamp){
202  fprintf(stdout, "%d: %s\n", (int)now, s);
203  }else{
204  fprintf(stdout, "%s\n", s);
205  }
206  fflush(stdout);
207  }
208  if(log_destinations & MQTT3_LOG_STDERR){
209  if(int_db.config && int_db.config->log_timestamp){
210  fprintf(stderr, "%d: %s\n", (int)now, s);
211  }else{
212  fprintf(stderr, "%s\n", s);
213  }
214  fflush(stderr);
215  }
216  if(log_destinations & MQTT3_LOG_FILE && int_db.config->log_fptr){
217  if(int_db.config && int_db.config->log_timestamp){
218  fprintf(int_db.config->log_fptr, "%d: %s\n", (int)now, s);
219  }else{
220  fprintf(int_db.config->log_fptr, "%s\n", s);
221  }
222  if(now - last_flush > 1){
223  fflush(int_db.config->log_fptr);
224  last_flush = now;
225  }
226  }
227  if(log_destinations & MQTT3_LOG_SYSLOG){
228 #ifndef WIN32
229  syslog(syslog_priority, "%s", s);
230 #else
231  sp = (char *)s;
232  ReportEvent(syslog_h, syslog_priority, 0, 0, NULL, 1, 0, &sp, NULL);
233 #endif
234  }
235  if(log_destinations & MQTT3_LOG_TOPIC && priority != MOSQ_LOG_DEBUG){
236  if(int_db.config && int_db.config->log_timestamp){
237  len += 30;
238  st = _mosquitto_malloc(len*sizeof(char));
239  if(!st){
240  _mosquitto_free(s);
241  return MOSQ_ERR_NOMEM;
242  }
243  snprintf(st, len, "%d: %s", (int)now, s);
244  mqtt3_db_messages_easy_queue(&int_db, NULL, topic, 2, strlen(st), st, 0);
245  _mosquitto_free(st);
246  }else{
247  mqtt3_db_messages_easy_queue(&int_db, NULL, topic, 2, strlen(s), s, 0);
248  }
249  }
250  _mosquitto_free(s);
251  }
252 
253  return MOSQ_ERR_SUCCESS;
254 }
