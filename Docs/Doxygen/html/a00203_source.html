218 int mosquitto_acl_check(struct mosquitto_db *db, struct mosquitto *context, const char *topic, int access)
219 {
220  char *username;
221 
222  if(!context->id){
223  return MOSQ_ERR_ACL_DENIED;
224  }
225  if(!db->auth_plugin.lib){
226  return mosquitto_acl_check_default(db, context, topic, access);
227  }else{
228 #ifdef WITH_BRIDGE
229  if(context->bridge){
230  username = context->bridge->local_username;
231  }else
232 #endif
233  {
234  username = context->username;
235  }
236 
237  if(db->config->auth_plugin_deny_special_chars == true){
238  /* Check whether the client id or username contains a + or # and if
239  * so deny access.
240  *
241  * Do this check for every message regardless, we have to protect the
242  * plugins against possible pattern based attacks.
243  */
244  if(username && strpbrk(username, "+#")){
245  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "ACL denying access to client with dangerous username \"%s\"", username);
246  return MOSQ_ERR_ACL_DENIED;
247  }
248  if(context->id && strpbrk(context->id, "+#")){
249  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "ACL denying access to client with dangerous client id \"%s\"", context->id);
250  return MOSQ_ERR_ACL_DENIED;
251  }
252  }
253  return db->auth_plugin.acl_check(db->auth_plugin.user_data, context->id, username, topic, access);
254  }
255 }
