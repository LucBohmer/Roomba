44 int base64_encode(unsigned char *in, unsigned int in_len, char **encoded)
45 {
46  BIO *bmem, *b64;
47  BUF_MEM *bptr;
48 
49  b64 = BIO_new(BIO_f_base64());
50  BIO_set_flags(b64, BIO_FLAGS_BASE64_NO_NL);
51  bmem = BIO_new(BIO_s_mem());
52  b64 = BIO_push(b64, bmem);
53  BIO_write(b64, in, in_len);
54  if(BIO_flush(b64) != 1){
55  BIO_free_all(b64);
56  return 1;
57  }
58  BIO_get_mem_ptr(b64, &bptr);
59  *encoded = malloc(bptr->length+1);
60  if(!(*encoded)){
61  BIO_free_all(b64);
62  return 1;
63  }
64  memcpy(*encoded, bptr->data, bptr->length);
65  (*encoded)[bptr->length] = '\0';
66  BIO_free_all(b64);
67 
68  return 0;
69 }

85 int output_new_password(FILE *fptr, const char *username, const char *password)
86 {
87  int rc;
88  unsigned char salt[SALT_LEN];
89  char *salt64 = NULL, *hash64 = NULL;
90  unsigned char hash[EVP_MAX_MD_SIZE];
91  unsigned int hash_len;
92  const EVP_MD *digest;
93 #if OPENSSL_VERSION_NUMBER < 0x10100000L
94  EVP_MD_CTX context;
95 #else
96  EVP_MD_CTX *context;
97 #endif
98 
99  rc = RAND_bytes(salt, SALT_LEN);
100  if(!rc){
101  fprintf(stderr, "Error: Insufficient entropy available to perform password generation.\n");
102  return 1;
103  }
104 
105  rc = base64_encode(salt, SALT_LEN, &salt64);
106  if(rc){
107  if(salt64) free(salt64);
108  fprintf(stderr, "Error: Unable to encode salt.\n");
109  return 1;
110  }
111 
112 
113  digest = EVP_get_digestbyname("sha512");
114  if(!digest){
115  if(salt64) free(salt64);
116  fprintf(stderr, "Error: Unable to create openssl digest.\n");
117  return 1;
118  }
119 
120 #if OPENSSL_VERSION_NUMBER < 0x10100000L
121  EVP_MD_CTX_init(&context);
122  EVP_DigestInit_ex(&context, digest, NULL);
123  EVP_DigestUpdate(&context, password, strlen(password));
124  EVP_DigestUpdate(&context, salt, SALT_LEN);
125  EVP_DigestFinal_ex(&context, hash, &hash_len);
126  EVP_MD_CTX_cleanup(&context);
127 #else
128  context = EVP_MD_CTX_new();
129  EVP_DigestInit_ex(context, digest, NULL);
130  EVP_DigestUpdate(context, password, strlen(password));
131  EVP_DigestUpdate(context, salt, SALT_LEN);
132  EVP_DigestFinal_ex(context, hash, &hash_len);
133  EVP_MD_CTX_free(context);
134 #endif
135 
136  rc = base64_encode(hash, hash_len, &hash64);
137  if(rc){
138  if(salt64) free(salt64);
139  if(hash64) free(hash64);
140  fprintf(stderr, "Error: Unable to encode hash.\n");
141  return 1;
142  }
143 
144  fprintf(fptr, "%s:$6$%s$%s\n", username, salt64, hash64);
145  free(salt64);
146  free(hash64);
147 
148  return 0;
149 }
