77 /* mosquitto shouldn't run as root.
78  * This function will attempt to change to an unprivileged user and group if
79  * running as root. The user is given in config->user.
80  * Returns 1 on failure (unknown user, setuid/setgid failure)
81  * Returns 0 on success.
82  * Note that setting config->user to "root" does not produce an error, but it
83  * strongly discouraged.
84  */
85 int drop_privileges(struct mqtt3_config *config, bool temporary)
86 {
87 #if !defined(__CYGWIN__) && !defined(WIN32)
88  struct passwd *pwd;
89  char err[256];
90  int rc;
91 
92  if(geteuid() == 0){
93  if(config->user && strcmp(config->user, "root")){
94  pwd = getpwnam(config->user);
95  if(!pwd){
96  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Invalid user '%s'.", config->user);
97  return 1;
98  }
99  if(initgroups(config->user, pwd->pw_gid) == -1){
100  strerror_r(errno, err, 256);
101  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error setting groups whilst dropping privileges: %s.", err);
102  return 1;
103  }
104  if(temporary){
105  rc = setegid(pwd->pw_gid);
106  }else{
107  rc = setgid(pwd->pw_gid);
108  }
109  if(rc == -1){
110  strerror_r(errno, err, 256);
111  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error setting gid whilst dropping privileges: %s.", err);
112  return 1;
113  }
114  if(temporary){
115  rc = seteuid(pwd->pw_uid);
116  }else{
117  rc = setuid(pwd->pw_uid);
118  }
119  if(rc == -1){
120  strerror_r(errno, err, 256);
121  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error setting uid whilst dropping privileges: %s.", err);
122  return 1;
123  }
124  }
125  if(geteuid() == 0 || getegid() == 0){
126  _mosquitto_log_printf(NULL, MOSQ_LOG_WARNING, "Warning: Mosquitto should not be run as root/administrator.");
127  }
128  }
129 #endif
130  return MOSQ_ERR_SUCCESS;
131 }

179 void mosquitto__daemonise(void)
180 {
181 #ifndef WIN32
182  char err[256];
183  pid_t pid;
184 
185  pid = fork();
186  if(pid < 0){
187  strerror_r(errno, err, 256);
188  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error in fork: %s", err);
189  exit(1);
190  }
191  if(pid > 0){
192  exit(0);
193  }
194  if(setsid() < 0){
195  strerror_r(errno, err, 256);
196  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error in setsid: %s", err);
197  exit(1);
198  }
199 
200  assert(freopen("/dev/null", "r", stdin));
201  assert(freopen("/dev/null", "w", stdout));
202  assert(freopen("/dev/null", "w", stderr));
203 #else
204  _mosquitto_log_printf(NULL, MOSQ_LOG_WARNING, "Warning: Can't start in daemon mode in Windows.");
205 #endif
206 }
