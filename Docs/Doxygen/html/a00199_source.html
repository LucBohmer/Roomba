86 int mqtt3_socket_accept(struct mosquitto_db *db, mosq_sock_t listensock)
87 {
88  int i;
89  int j;
90  mosq_sock_t new_sock = INVALID_SOCKET;
91  struct mosquitto *new_context;
92 #ifdef WITH_TLS
93  BIO *bio;
94  int rc;
95  char ebuf[256];
96  unsigned long e;
97 #endif
98 #ifdef WITH_WRAP
99  struct request_info wrap_req;
100  char address[1024];
101 #endif
102 
103  new_sock = accept(listensock, NULL, 0);
104  if(new_sock == INVALID_SOCKET) return -1;
105 
106 #ifdef WITH_SYS_TREE
107  g_socket_connections++;
108 #endif
109 
110  if(_mosquitto_socket_nonblock(new_sock)){
111  return INVALID_SOCKET;
112  }
113 
114 #ifdef WITH_WRAP
115  /* Use tcpd / libwrap to determine whether a connection is allowed. */
116  request_init(&wrap_req, RQ_FILE, new_sock, RQ_DAEMON, "mosquitto", 0);
117  fromhost(&wrap_req);
118  if(!hosts_access(&wrap_req)){
119  /* Access is denied */
120  if(!_mosquitto_socket_get_address(new_sock, address, 1024)){
121  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "Client connection from %s denied access by tcpd.", address);
122  }
123  COMPAT_CLOSE(new_sock);
124  return -1;
125  }
126 #endif
127  new_context = mqtt3_context_init(db, new_sock);
128  if(!new_context){
129  COMPAT_CLOSE(new_sock);
130  return -1;
131  }
132  for(i=0; i<db->config->listener_count; i++){
133  for(j=0; j<db->config->listeners[i].sock_count; j++){
134  if(db->config->listeners[i].socks[j] == listensock){
135  new_context->listener = &db->config->listeners[i];
136  new_context->listener->client_count++;
137  break;
138  }
139  }
140  }
141  if(!new_context->listener){
142  mqtt3_context_cleanup(db, new_context, true);
143  return -1;
144  }
145 
146  if(new_context->listener->max_connections > 0 && new_context->listener->client_count > new_context->listener->max_connections){
147  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "Client connection from %s denied: max_connections exceeded.", new_context->address);
148  mqtt3_context_cleanup(db, new_context, true);
149  return -1;
150  }
151 
152 #ifdef WITH_TLS
153  /* TLS init */
154  for(i=0; i<db->config->listener_count; i++){
155  for(j=0; j<db->config->listeners[i].sock_count; j++){
156  if(db->config->listeners[i].socks[j] == listensock){
157  if(db->config->listeners[i].ssl_ctx){
158  new_context->ssl = SSL_new(db->config->listeners[i].ssl_ctx);
159  if(!new_context->ssl){
160  mqtt3_context_cleanup(db, new_context, true);
161  return -1;
162  }
163  SSL_set_ex_data(new_context->ssl, tls_ex_index_context, new_context);
164  SSL_set_ex_data(new_context->ssl, tls_ex_index_listener, &db->config->listeners[i]);
165  new_context->want_write = true;
166  bio = BIO_new_socket(new_sock, BIO_NOCLOSE);
167  SSL_set_bio(new_context->ssl, bio, bio);
168  ERR_clear_error();
169  rc = SSL_accept(new_context->ssl);
170  if(rc != 1){
171  rc = SSL_get_error(new_context->ssl, rc);
172  if(rc == SSL_ERROR_WANT_READ){
173  /* We always want to read. */
174  }else if(rc == SSL_ERROR_WANT_WRITE){
175  new_context->want_write = true;
176  }else{
177  e = ERR_get_error();
178  while(e){
179  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE,
180  "Client connection from %s failed: %s.",
181  new_context->address, ERR_error_string(e, ebuf));
182  e = ERR_get_error();
183  }
184  mqtt3_context_cleanup(db, new_context, true);
185  return -1;
186  }
187  }
188  }
189  }
190  }
191  }
192 #endif
193 
194  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "New connection from %s on port %d.", new_context->address, new_context->listener->port);
195 
196  return new_sock;
197 }

343 int mqtt3_socket_listen(struct _mqtt3_listener *listener)
344 {
345  mosq_sock_t sock = INVALID_SOCKET;
346  struct addrinfo hints;
347  struct addrinfo *ainfo, *rp;
348  char service[10];
349 #ifndef WIN32
350  int ss_opt = 1;
351 #else
352  char ss_opt = 1;
353 #endif
354 #ifdef WITH_TLS
355  int rc;
356  X509_STORE *store;
357  X509_LOOKUP *lookup;
358 #endif
359 
360  if(!listener) return MOSQ_ERR_INVAL;
361 
362  snprintf(service, 10, "%d", listener->port);
363  memset(&hints, 0, sizeof(struct addrinfo));
364  hints.ai_family = PF_UNSPEC;
365  hints.ai_flags = AI_PASSIVE;
366  hints.ai_socktype = SOCK_STREAM;
367 
368  if(getaddrinfo(listener->host, service, &hints, &ainfo)) return INVALID_SOCKET;
369 
370  listener->sock_count = 0;
371  listener->socks = NULL;
372 
373  for(rp = ainfo; rp; rp = rp->ai_next){
374  if(rp->ai_family == AF_INET){
375  _mosquitto_log_printf(NULL, MOSQ_LOG_INFO, "Opening ipv4 listen socket on port %d.", ntohs(((struct sockaddr_in *)rp->ai_addr)->sin_port));
376  }else if(rp->ai_family == AF_INET6){
377  _mosquitto_log_printf(NULL, MOSQ_LOG_INFO, "Opening ipv6 listen socket on port %d.", ntohs(((struct sockaddr_in6 *)rp->ai_addr)->sin6_port));
378  }else{
379  continue;
380  }
381 
382  sock = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
383  if(sock == INVALID_SOCKET){
384  net__print_error(MOSQ_LOG_WARNING, "Warning: %s");
385  continue;
386  }
387  listener->sock_count++;
388  listener->socks = _mosquitto_realloc(listener->socks, sizeof(mosq_sock_t)*listener->sock_count);
389  if(!listener->socks){
390  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Out of memory.");
391  return MOSQ_ERR_NOMEM;
392  }
393  listener->socks[listener->sock_count-1] = sock;
394 
395 #ifndef WIN32
396  ss_opt = 1;
397  setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &ss_opt, sizeof(ss_opt));
398 #endif
399  ss_opt = 1;
400  setsockopt(sock, IPPROTO_IPV6, IPV6_V6ONLY, &ss_opt, sizeof(ss_opt));
401 
402  if(_mosquitto_socket_nonblock(sock)){
403  return 1;
404  }
405 
406  if(bind(sock, rp->ai_addr, rp->ai_addrlen) == -1){
407  net__print_error(MOSQ_LOG_ERR, "Error: %s");
408  COMPAT_CLOSE(sock);
409  return 1;
410  }
411 
412  if(listen(sock, 100) == -1){
413  net__print_error(MOSQ_LOG_ERR, "Error: %s");
414  COMPAT_CLOSE(sock);
415  return 1;
416  }
417  }
418  freeaddrinfo(ainfo);
419 
420  /* We need to have at least one working socket. */
421  if(listener->sock_count > 0){
422 #ifdef WITH_TLS
423  if((listener->cafile || listener->capath) && listener->certfile && listener->keyfile){
424  if(_mosquitto_tls_server_ctx(listener)){
425  COMPAT_CLOSE(sock);
426  return 1;
427  }
428 
429  rc = SSL_CTX_load_verify_locations(listener->ssl_ctx, listener->cafile, listener->capath);
430  if(rc == 0){
431  if(listener->cafile && listener->capath){
432  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to load CA certificates. Check cafile \"%s\" and capath \"%s\".", listener->cafile, listener->capath);
433  }else if(listener->cafile){
434  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to load CA certificates. Check cafile \"%s\".", listener->cafile);
435  }else{
436  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to load CA certificates. Check capath \"%s\".", listener->capath);
437  }
438  COMPAT_CLOSE(sock);
439  return 1;
440  }
441  /* FIXME user data? */
442  if(listener->require_certificate){
443  SSL_CTX_set_verify(listener->ssl_ctx, SSL_VERIFY_PEER | SSL_VERIFY_FAIL_IF_NO_PEER_CERT, client_certificate_verify);
444  }else{
445  SSL_CTX_set_verify(listener->ssl_ctx, SSL_VERIFY_NONE, client_certificate_verify);
446  }
447  rc = SSL_CTX_use_certificate_chain_file(listener->ssl_ctx, listener->certfile);
448  if(rc != 1){
449  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to load server certificate \"%s\". Check certfile.", listener->certfile);
450  COMPAT_CLOSE(sock);
451  return 1;
452  }
453  rc = SSL_CTX_use_PrivateKey_file(listener->ssl_ctx, listener->keyfile, SSL_FILETYPE_PEM);
454  if(rc != 1){
455  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to load server key file \"%s\". Check keyfile.", listener->keyfile);
456  COMPAT_CLOSE(sock);
457  return 1;
458  }
459  rc = SSL_CTX_check_private_key(listener->ssl_ctx);
460  if(rc != 1){
461  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Server certificate/key are inconsistent.");
462  COMPAT_CLOSE(sock);
463  return 1;
464  }
465  /* Load CRLs if they exist. */
466  if(listener->crlfile){
467  store = SSL_CTX_get_cert_store(listener->ssl_ctx);
468  if(!store){
469  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to obtain TLS store.");
470  COMPAT_CLOSE(sock);
471  return 1;
472  }
473  lookup = X509_STORE_add_lookup(store, X509_LOOKUP_file());
474  rc = X509_load_crl_file(lookup, listener->crlfile, X509_FILETYPE_PEM);
475  if(rc != 1){
476  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to load certificate revocation file \"%s\". Check crlfile.", listener->crlfile);
477  COMPAT_CLOSE(sock);
478  return 1;
479  }
480  X509_STORE_set_flags(store, X509_V_FLAG_CRL_CHECK);
481  }
482 
483 # ifdef REAL_WITH_TLS_PSK
484  }else if(listener->psk_hint){
485  if(tls_ex_index_context == -1){
486  tls_ex_index_context = SSL_get_ex_new_index(0, "client context", NULL, NULL, NULL);
487  }
488  if(tls_ex_index_listener == -1){
489  tls_ex_index_listener = SSL_get_ex_new_index(0, "listener", NULL, NULL, NULL);
490  }
491 
492  if(_mosquitto_tls_server_ctx(listener)){
493  COMPAT_CLOSE(sock);
494  return 1;
495  }
496  SSL_CTX_set_psk_server_callback(listener->ssl_ctx, psk_server_callback);
497  if(listener->psk_hint){
498  rc = SSL_CTX_use_psk_identity_hint(listener->ssl_ctx, listener->psk_hint);
499  if(rc == 0){
500  _mosquitto_log_printf(NULL, MOSQ_LOG_ERR, "Error: Unable to set TLS PSK hint.");
501  COMPAT_CLOSE(sock);
502  return 1;
503  }
504  }
505 # endif /* REAL_WITH_TLS_PSK */
506  }
507 #endif /* WITH_TLS */
508  return 0;
509  }else{
510  return 1;
511  }
512 }
