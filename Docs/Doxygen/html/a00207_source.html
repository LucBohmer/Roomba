17 /* A note on matching topic subscriptions.
18  *
19  * Topics can be up to 32767 characters in length. The / character is used as a
20  * hierarchy delimiter. Messages are published to a particular topic.
21  * Clients may subscribe to particular topics directly, but may also use
22  * wildcards in subscriptions. The + and # characters are used as wildcards.
23  * The # wildcard can be used at the end of a subscription only, and is a
24  * wildcard for the level of hierarchy at which it is placed and all subsequent
25  * levels.
26  * The + wildcard may be used at any point within the subscription and is a
27  * wildcard for only the level of hierarchy at which it is placed.
28  * Neither wildcard may be used as part of a substring.
29  * Valid:
30  * a/b/+
31  * a/+/c
32  * a/#
33  * a/b/#
34  * #
35  * +/b/c
36  * +/+/+
37  * Invalid:
38  * a/#/c
39  * a+/b/c
40  * Valid but non-matching:
41  * a/b
42  * a/+
43  * +/b
44  * b/c/a
45  * a/b/d
46  */

67  int client_qos, msg_qos;
68  uint16_t mid;
69  struct _mosquitto_subleaf *leaf;
70  bool client_retain;
71 
72  leaf = hier->subs;
73 
74  if(retain && set_retain){
75 #ifdef WITH_PERSISTENCE
76  if(strncmp(topic, "$SYS", 4)){
77  /* Retained messages count as a persistence change, but only if
78  * they aren't for $SYS. */

94  }else{
95  hier->retained = NULL;
96  }
97  }

99  if(!leaf->context->id || (leaf->context->is_bridge && !strcmp(leaf->context->id, source_id))){
100  leaf = leaf->next;
101  continue;
102  }
103  /* Check for ACL topic access. */
104  rc2 = mosquitto_acl_check(db, leaf->context, topic, MOSQ_ACL_READ);
105  if(rc2 == MOSQ_ERR_ACL_DENIED){
106  leaf = leaf->next;
107  continue;
108  }else if(rc2 == MOSQ_ERR_SUCCESS){
109  client_qos = leaf->qos;
110 
111  if(db->config->upgrade_outgoing_qos){
112  msg_qos = client_qos;
113  }else{
114  if(qos > client_qos){
115  msg_qos = client_qos;
116  }else{
117  msg_qos = qos;
118  }
119  }
120  if(msg_qos){
121  mid = _mosquitto_mid_generate(leaf->context);
122  }else{
123  mid = 0;
124  }
125  if(leaf->context->is_bridge){
126  /* If we know the client is a bridge then we should set retain
127  * even if the message is fresh. If we don't do this, retained
128  * messages won't be propagated. */
129  client_retain = retain;
130  }else{
131  /* Client is not a bridge and this isn't a stale message so
132  * retain should be false. */
133  client_retain = false;
134  }
135  if(mqtt3_db_message_insert(db, leaf->context, mid, mosq_md_out, msg_qos, client_retain, stored) == 1) rc = 1;
136  }else{
137  return 1; /* Application error */
138  }
139  leaf = leaf->next;
140  }
141  return rc;
142 }

169 static int _sub_topic_tokenise(const char *subtopic, struct _sub_token **topics)
170 {
171  struct _sub_token *new_topic, *tail = NULL;
172  int len;
173  int start, stop, tlen;
174  int i;
175  char *topic;
176 
177  assert(subtopic);
178  assert(topics);
179 
180  if(subtopic[0] != '$'){
181  new_topic = _sub_topic_append(&tail, topics, _mosquitto_strdup(""));
182  if(!new_topic) goto cleanup;
183  }
184 
185  len = strlen(subtopic);
186 
187  if(subtopic[0] == '/'){
188  new_topic = _sub_topic_append(&tail, topics, _mosquitto_strdup(""));
189  if(!new_topic) goto cleanup;
190 
191  start = 1;
192  }else{
193  start = 0;
194  }
195 
196  stop = 0;
197  for(i=start; i<len+1; i++){
198  if(subtopic[i] == '/' || subtopic[i] == '\0'){
199  stop = i;
200 
201  if(start != stop){
202  tlen = stop-start;
203 
204  topic = _mosquitto_malloc(tlen+1);
205  if(!topic) goto cleanup;
206  memcpy(topic, &subtopic[start], tlen);
207  topic[tlen] = '\0';
208  }else{
209  topic = _mosquitto_strdup("");
210  if(!topic) goto cleanup;
211  }
212  new_topic = _sub_topic_append(&tail, topics, topic);
213  if(!new_topic) goto cleanup;
214  start = i+1;
215  }
216  }
217 
218  return MOSQ_ERR_SUCCESS;
219 
220 cleanup:
221  tail = *topics;
222  *topics = NULL;
223  while(tail){
224  if(tail->topic) _mosquitto_free(tail->topic);
225  new_topic = tail->next;
226  _mosquitto_free(tail);
227  tail = new_topic;
228  }
229  return 1;
230 }

397 {
398  /* FIXME - need to take into account source_id if the client is a bridge */
399  struct _mosquitto_subhier *branch;
400  bool sr;
401 
402  branch = subhier->children;
403  while(branch){
404  sr = set_retain;
405 
406  if(tokens && tokens->topic && (!strcmp(branch->topic, tokens->topic) || !strcmp(branch->topic, "+"))){
407  /* The topic matches this subscription.
408  * Doesn't include # wildcards */
409  if(!strcmp(branch->topic, "+")){
410  /* Don't set a retained message where + is in the hierarchy. */
411  sr = false;
412  }
413  _sub_search(db, branch, tokens->next, source_id, topic, qos, retain, stored, sr);

420  * there may still be other subscriptions to deal with.
