109 static void calc_load(struct mosquitto_db *db, char *buf, const char *topic, double exponent, double interval, double *current)
110 {
111  double new_value;
112 
113  new_value = interval + exponent*((*current) - interval);

115  snprintf(buf, BUFLEN, "%.2f", new_value);
116  mqtt3_db_messages_easy_queue(db, NULL, topic, 2, strlen(buf), buf, 1);
117  }
118  (*current) = new_value;
119 }
