34 int mqtt3_packet_handle(struct mosquitto_db *db, struct mosquitto *context)
35 {
36  if(!context) return MOSQ_ERR_INVAL;
37 
38  switch((context->in_packet.command)&0xF0){
39  case PINGREQ:
40  return _mosquitto_handle_pingreq(context);
41  case PINGRESP:
42  return _mosquitto_handle_pingresp(context);
43  case PUBACK:
44  return _mosquitto_handle_pubackcomp(db, context, "PUBACK");
45  case PUBCOMP:
46  return _mosquitto_handle_pubackcomp(db, context, "PUBCOMP");
47  case PUBLISH:
48  return mqtt3_handle_publish(db, context);
49  case PUBREC:
50  return _mosquitto_handle_pubrec(context);
51  case PUBREL:
52  return _mosquitto_handle_pubrel(db, context);
53  case CONNECT:
54  return mqtt3_handle_connect(db, context);
55  case DISCONNECT:
56  return mqtt3_handle_disconnect(db, context);
57  case SUBSCRIBE:
58  return mqtt3_handle_subscribe(db, context);
59  case UNSUBSCRIBE:
60  return mqtt3_handle_unsubscribe(db, context);
61 #ifdef WITH_BRIDGE
62  case CONNACK:
63  return mqtt3_handle_connack(db, context);
64  case SUBACK:
65  return _mosquitto_handle_suback(context);
66  case UNSUBACK:
67  return _mosquitto_handle_unsuback(context);
68 #endif
69  default:
70  /* If we don't recognise the command, return an error straight away. */
71  return MOSQ_ERR_PROTOCOL;
72  }
73 }

75 int mqtt3_handle_publish(struct mosquitto_db *db, struct mosquitto *context)
76 {
77  char *topic;
78  void *payload = NULL;
79  uint32_t payloadlen;
80  uint8_t dup, qos, retain;
81  uint16_t mid = 0;
82  int rc = 0;
83  uint8_t header = context->in_packet.command;
84  int res = 0;
85  struct mosquitto_msg_store *stored = NULL;
86  int len;
87  char *topic_mount;
88 #ifdef WITH_BRIDGE
89  char *topic_temp;
90  int i;
91  struct _mqtt3_bridge_topic *cur_topic;
92  bool match;
93 #endif
94 
95  dup = (header & 0x08)>>3;
96  qos = (header & 0x06)>>1;
97  if(qos == 3){
98  _mosquitto_log_printf(NULL, MOSQ_LOG_INFO,
99  "Invalid QoS in PUBLISH from %s, disconnecting.", context->id);
100  return 1;
101  }
102  retain = (header & 0x01);
103 
104  if(_mosquitto_read_string(&context->in_packet, &topic)) return 1;
105  if(STREMPTY(topic)){
106  /* Invalid publish topic, disconnect client. */
107  _mosquitto_free(topic);
108  return 1;
109  }
110 #ifdef WITH_BRIDGE
111  if(context->bridge && context->bridge->topics && context->bridge->topic_remapping){
112  for(i=0; i<context->bridge->topic_count; i++){
113  cur_topic = &context->bridge->topics[i];
114  if((cur_topic->direction == bd_both || cur_topic->direction == bd_in)
115  && (cur_topic->remote_prefix || cur_topic->local_prefix)){
116 
117  /* Topic mapping required on this topic if the message matches */
118 
119  rc = mosquitto_topic_matches_sub(cur_topic->remote_topic, topic, &match);
120  if(rc){
121  _mosquitto_free(topic);
122  return rc;
123  }
124  if(match){
125  if(cur_topic->remote_prefix){
126  /* This prefix needs removing. */
127  if(!strncmp(cur_topic->remote_prefix, topic, strlen(cur_topic->remote_prefix))){
128  topic_temp = _mosquitto_strdup(topic+strlen(cur_topic->remote_prefix));
129  if(!topic_temp){
130  _mosquitto_free(topic);
131  return MOSQ_ERR_NOMEM;
132  }
133  _mosquitto_free(topic);
134  topic = topic_temp;
135  }
136  }
137 
138  if(cur_topic->local_prefix){
139  /* This prefix needs adding. */
140  len = strlen(topic) + strlen(cur_topic->local_prefix)+1;
141  topic_temp = _mosquitto_malloc(len+1);
142  if(!topic_temp){
143  _mosquitto_free(topic);
144  return MOSQ_ERR_NOMEM;
145  }
146  snprintf(topic_temp, len, "%s%s", cur_topic->local_prefix, topic);
147  topic_temp[len] = '\0';
148 
149  _mosquitto_free(topic);
150  topic = topic_temp;
151  }
152  break;
153  }
154  }
155  }
156  }
157 #endif
158  if(mosquitto_pub_topic_check(topic) != MOSQ_ERR_SUCCESS){
159  /* Invalid publish topic, just swallow it. */
160  _mosquitto_free(topic);
161  return 1;
162  }
163 
164  if(qos > 0){
165  if(_mosquitto_read_uint16(&context->in_packet, &mid)){
166  _mosquitto_free(topic);
167  return 1;
168  }
169  }
170 
171  payloadlen = context->in_packet.remaining_length - context->in_packet.pos;
172 #ifdef WITH_SYS_TREE
173  g_pub_bytes_received += payloadlen;
174 #endif
175  if(context->listener && context->listener->mount_point){
176  len = strlen(context->listener->mount_point) + strlen(topic) + 1;
177  topic_mount = _mosquitto_malloc(len+1);
178  if(!topic_mount){
179  _mosquitto_free(topic);
180  return MOSQ_ERR_NOMEM;
181  }
182  snprintf(topic_mount, len, "%s%s", context->listener->mount_point, topic);
183  topic_mount[len] = '\0';
184 
185  _mosquitto_free(topic);
186  topic = topic_mount;
187  }
188 
189  if(payloadlen){
190  if(db->config->message_size_limit && payloadlen > db->config->message_size_limit){
191  _mosquitto_log_printf(NULL, MOSQ_LOG_DEBUG, "Dropped too large PUBLISH from %s (d%d, q%d, r%d, m%d, '%s', ... (%ld bytes))", context->id, dup, qos, retain, mid, topic, (long)payloadlen);
192  goto process_bad_message;
193  }
194  payload = _mosquitto_calloc(payloadlen+1, 1);
195  if(!payload){
196  _mosquitto_free(topic);
197  return 1;
198  }
199  if(_mosquitto_read_bytes(&context->in_packet, payload, payloadlen)){
200  _mosquitto_free(topic);
201  _mosquitto_free(payload);
202  return 1;
203  }
204  }
205 
206  /* Check for topic access */
207  rc = mosquitto_acl_check(db, context, topic, MOSQ_ACL_WRITE);
208  if(rc == MOSQ_ERR_ACL_DENIED){
209  _mosquitto_log_printf(NULL, MOSQ_LOG_DEBUG, "Denied PUBLISH from %s (d%d, q%d, r%d, m%d, '%s', ... (%ld bytes))", context->id, dup, qos, retain, mid, topic, (long)payloadlen);
210  goto process_bad_message;
211  }else if(rc != MOSQ_ERR_SUCCESS){
212  _mosquitto_free(topic);
213  if(payload) _mosquitto_free(payload);
214  return rc;
215  }
216 
217  _mosquitto_log_printf(NULL, MOSQ_LOG_DEBUG, "Received PUBLISH from %s (d%d, q%d, r%d, m%d, '%s', ... (%ld bytes))", context->id, dup, qos, retain, mid, topic, (long)payloadlen);
218  if(qos > 0){
219  mqtt3_db_message_store_find(context, mid, &stored);
220  }
221  if(!stored){
222  dup = 0;
223  if(mqtt3_db_message_store(db, context->id, mid, topic, qos, payloadlen, payload, retain, &stored, 0)){
224  _mosquitto_free(topic);
225  if(payload) _mosquitto_free(payload);
226  return 1;
227  }
228  }else{
229  dup = 1;
230  }
231  switch(qos){
232  case 0:
233  if(mqtt3_db_messages_queue(db, context->id, topic, qos, retain, &stored)) rc = 1;
234  break;
235  case 1:
236  if(mqtt3_db_messages_queue(db, context->id, topic, qos, retain, &stored)) rc = 1;
237  if(_mosquitto_send_puback(context, mid)) rc = 1;
238  break;
239  case 2:
240  if(!dup){
241  res = mqtt3_db_message_insert(db, context, mid, mosq_md_in, qos, retain, stored);
242  }else{
243  res = 0;
244  }
245  /* mqtt3_db_message_insert() returns 2 to indicate dropped message
246  * due to queue. This isn't an error so don't disconnect them. */
247  if(!res){
248  if(_mosquitto_send_pubrec(context, mid)) rc = 1;
249  }else if(res == 1){
250  rc = 1;
251  }
252  break;
253  }
254  _mosquitto_free(topic);
255  if(payload) _mosquitto_free(payload);
256 
257  return rc;
258 process_bad_message:
259  _mosquitto_free(topic);
260  if(payload) _mosquitto_free(payload);
261  switch(qos){
262  case 0:
263  return MOSQ_ERR_SUCCESS;
264  case 1:
265  return _mosquitto_send_puback(context, mid);
266  case 2:
267  mqtt3_db_message_store_find(context, mid, &stored);
268  if(!stored){
269  if(mqtt3_db_message_store(db, context->id, mid, NULL, qos, 0, NULL, false, &stored, 0)){
270  return 1;
271  }
272  res = mqtt3_db_message_insert(db, context, mid, mosq_md_in, qos, false, stored);
273  }else{
274  res = 0;
275  }
276  if(!res){
277  res = _mosquitto_send_pubrec(context, mid);
278  }
279  return res;
280  }
281  return 1;
282 }
