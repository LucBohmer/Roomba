46 int _mosquitto_packet_alloc(struct _mosquitto_packet *packet)
47 {
48  uint8_t remaining_bytes[5], byte;
49  uint32_t remaining_length;
50  int i;
51 
52  assert(packet);
53 
54  remaining_length = packet->remaining_length;
55  packet->payload = NULL;
56  packet->remaining_count = 0;
57  do{
58  byte = remaining_length % 128;
59  remaining_length = remaining_length / 128;
60  /* If there are more digits to encode, set the top bit of this digit */
61  if(remaining_length > 0){
62  byte = byte | 0x80;
63  }
64  remaining_bytes[packet->remaining_count] = byte;
65  packet->remaining_count++;
66  }while(remaining_length > 0 && packet->remaining_count < 5);
67  if(packet->remaining_count == 5) return MOSQ_ERR_PAYLOAD_SIZE;
68  packet->packet_length = packet->remaining_length + 1 + packet->remaining_count;
69 #ifdef WITH_WEBSOCKETS
70  packet->payload = _mosquitto_malloc(sizeof(uint8_t)*packet->packet_length + LWS_SEND_BUFFER_PRE_PADDING + LWS_SEND_BUFFER_POST_PADDING);
71 #else
72  packet->payload = _mosquitto_malloc(sizeof(uint8_t)*packet->packet_length);
73 #endif
74  if(!packet->payload) return MOSQ_ERR_NOMEM;
75 
76  packet->payload[0] = packet->command;
77  for(i=0; i<packet->remaining_count; i++){
78  packet->payload[i+1] = remaining_bytes[i];
79  }
80  packet->pos = 1 + packet->remaining_count;
81 
82  return MOSQ_ERR_SUCCESS;
83 }

87 #else
88 void _mosquitto_check_keepalive(struct mosquitto *mosq)
89 #endif
90 {
91  time_t next_msg_out;
92  time_t last_msg_in;
93  time_t now = mosquitto_time();
94 #ifndef WITH_BROKER
95  int rc;
96 #endif
97 
98  assert(mosq);
99 #if defined(WITH_BROKER) && defined(WITH_BRIDGE)
100  /* Check if a lazy bridge should be timed out due to idle. */
101  if(mosq->bridge && mosq->bridge->start_type == bst_lazy
102  && mosq->sock != INVALID_SOCKET
103  && now - mosq->next_msg_out - mosq->keepalive >= mosq->bridge->idle_timeout){
104 
105  _mosquitto_log_printf(NULL, MOSQ_LOG_NOTICE, "Bridge connection %s has exceeded idle timeout, disconnecting.", mosq->id);
106  _mosquitto_socket_close(db, mosq);
107  return;
108  }
109 #endif
110  pthread_mutex_lock(&mosq->msgtime_mutex);
111  next_msg_out = mosq->next_msg_out;
112  last_msg_in = mosq->last_msg_in;
113  pthread_mutex_unlock(&mosq->msgtime_mutex);
114  if(mosq->keepalive && mosq->sock != INVALID_SOCKET &&
115  (now >= next_msg_out || now - last_msg_in >= mosq->keepalive)){
116 
117  if(mosq->state == mosq_cs_connected && mosq->ping_t == 0){
118  _mosquitto_send_pingreq(mosq);
119  /* Reset last msg times to give the server time to send a pingresp */
120  pthread_mutex_lock(&mosq->msgtime_mutex);
121  mosq->last_msg_in = now;
122  mosq->next_msg_out = now + mosq->keepalive;
123  pthread_mutex_unlock(&mosq->msgtime_mutex);
124  }else{
125 #ifdef WITH_BROKER
126  if(mosq->listener){
127  mosq->listener->client_count--;
128  assert(mosq->listener->client_count >= 0);
129  }
130  mosq->listener = NULL;
131  _mosquitto_socket_close(db, mosq);
132 #else
133  _mosquitto_socket_close(mosq);
134  pthread_mutex_lock(&mosq->state_mutex);
135  if(mosq->state == mosq_cs_disconnecting){
136  rc = MOSQ_ERR_SUCCESS;
137  }else{
138  rc = 1;
139  }
140  pthread_mutex_unlock(&mosq->state_mutex);
141  pthread_mutex_lock(&mosq->callback_mutex);
142  if(mosq->on_disconnect){
143  mosq->in_callback = true;
144  mosq->on_disconnect(mosq, mosq->userdata, rc);
145  mosq->in_callback = false;
146  }
147  pthread_mutex_unlock(&mosq->callback_mutex);
148 #endif
149  }
150  }
151 }

154 {
155  /* FIXME - this would be better with atomic increment, but this is safer
156  * for now for a bug fix release.

161  */

194 /* Check that a topic used for subscriptions is valid.
195  * Search for + or # in a topic, check they aren't in invalid positions such as
196  * foo/#/bar, foo/+bar or foo/bar#.
197  * Return MOSQ_ERR_INVAL if invalid position found.
198  * Also returns MOSQ_ERR_INVAL if the topic string is too long.
199  * Returns MOSQ_ERR_SUCCESS if everything is fine.
200  */
201 int mosquitto_sub_topic_check(const char *str)
202 {
203  char c = '\0';
204  int len = 0;
205  while(str && str[0]){
206  if(str[0] == '+'){
207  if((c != '\0' && c != '/') || (str[1] != '\0' && str[1] != '/')){
208  return MOSQ_ERR_INVAL;
209  }
210  }else if(str[0] == '#'){
211  if((c != '\0' && c != '/') || str[1] != '\0'){
212  return MOSQ_ERR_INVAL;
213  }
214  }
215  len++;
216  c = str[0];
217  str = &str[1];
218  }
219  if(len > 65535) return MOSQ_ERR_INVAL;
220 
221  return MOSQ_ERR_SUCCESS;
222 }

224 /* Does a topic match a subscription? */
225 int mosquitto_topic_matches_sub(const char *sub, const char *topic, bool *result)
226 {
227  int slen, tlen;
228  int spos, tpos;
229  bool multilevel_wildcard = false;
230 
231  if(!sub || !topic || !result) return MOSQ_ERR_INVAL;
232 
233  slen = strlen(sub);
234  tlen = strlen(topic);
235 
236  if(!slen || !tlen){
237  *result = false;
238  return MOSQ_ERR_INVAL;
239  }
240 
241  if(slen && tlen){
242  if((sub[0] == '$' && topic[0] != '$')
243  || (topic[0] == '$' && sub[0] != '$')){
244 
245  *result = false;
246  return MOSQ_ERR_SUCCESS;
247  }
248  }
249 
250  spos = 0;
251  tpos = 0;
252 
253  while(spos < slen && tpos <= tlen){
254  if(sub[spos] == topic[tpos]){
255  if(tpos == tlen-1){
256  /* Check for e.g. foo matching foo/# */
257  if(spos == slen-3
258  && sub[spos+1] == '/'
259  && sub[spos+2] == '#'){
260  *result = true;
261  multilevel_wildcard = true;
262  return MOSQ_ERR_SUCCESS;
263  }
264  }
265  spos++;
266  tpos++;
267  if(spos == slen && tpos == tlen){
268  *result = true;
269  return MOSQ_ERR_SUCCESS;
270  }else if(tpos == tlen && spos == slen-1 && sub[spos] == '+'){
271  if(spos > 0 && sub[spos-1] != '/'){
272  *result = false;
273  return MOSQ_ERR_INVAL;
274  }
275  spos++;
276  *result = true;
277  return MOSQ_ERR_SUCCESS;
278  }
279  }else{
280  if(sub[spos] == '+'){
281  /* Check for bad "+foo" or "a/+foo" subscription */
282  if(spos > 0 && sub[spos-1] != '/'){
283  *result = false;
284  return MOSQ_ERR_INVAL;
285  }
286  /* Check for bad "foo+" or "foo+/a" subscription */
287  if(spos < slen-1 && sub[spos+1] != '/'){
288  *result = false;
289  return MOSQ_ERR_INVAL;
290  }
291  spos++;
292  while(tpos < tlen && topic[tpos] != '/'){
293  tpos++;
294  }
295  if(tpos == tlen && spos == slen){
296  *result = true;
297  return MOSQ_ERR_SUCCESS;
298  }
299  }else if(sub[spos] == '#'){
300  if(spos > 0 && sub[spos-1] != '/'){
301  *result = false;
302  return MOSQ_ERR_INVAL;
303  }
304  multilevel_wildcard = true;
305  if(spos+1 != slen){
306  *result = false;
307  return MOSQ_ERR_INVAL;
308  }else{
309  *result = true;
310  return MOSQ_ERR_SUCCESS;
311  }
312  }else{
313  *result = false;
314  return MOSQ_ERR_SUCCESS;
315  }
316  }
317  }
318  if(multilevel_wildcard == false && (tpos < tlen || spos < slen)){
319  *result = false;
320  }
321 
322  return MOSQ_ERR_SUCCESS;
323 }

346 FILE *_mosquitto_fopen(const char *path, const char *mode, bool restrict_read)
347 {
348 #ifdef WIN32
349  char buf[4096];
350  int rc;
351  rc = ExpandEnvironmentStrings(path, buf, 4096);
352  if(rc == 0 || rc > 4096){
353  return NULL;
354  }else{
355  if (restrict_read) {
356  HANDLE hfile;
357  SECURITY_ATTRIBUTES sec;
358  EXPLICIT_ACCESS ea;
359  PACL pacl = NULL;
360  char username[UNLEN + 1];
361  int ulen = UNLEN;
362  SECURITY_DESCRIPTOR sd;
363 
364  GetUserName(username, &ulen);
365  if (!InitializeSecurityDescriptor(&sd, SECURITY_DESCRIPTOR_REVISION)) {
366  return NULL;
367  }
368  BuildExplicitAccessWithName(&ea, username, GENERIC_ALL, SET_ACCESS, NO_INHERITANCE);
369  if (SetEntriesInAcl(1, &ea, NULL, &pacl) != ERROR_SUCCESS) {
370  return NULL;
371  }
372  if (!SetSecurityDescriptorDacl(&sd, TRUE, pacl, FALSE)) {
373  LocalFree(pacl);
374  return NULL;
375  }
376 
377  sec.nLength = sizeof(SECURITY_ATTRIBUTES);
378  sec.bInheritHandle = FALSE;
379  sec.lpSecurityDescriptor = &sd;
380 
381  hfile = CreateFile(buf, GENERIC_READ | GENERIC_WRITE, 0,
382  &sec,
383  CREATE_NEW,
384  FILE_ATTRIBUTE_NORMAL,
385  NULL);
386 
387  LocalFree(pacl);
388 
389  int fd = _open_osfhandle((intptr_t)hfile, 0);
390  if (fd < 0) {
391  return NULL;
392  }
393 
394  FILE *fptr = _fdopen(fd, mode);
395  if (!fptr) {
396  _close(fd);
397  return NULL;
398  }
399  return fptr;
400 
401  }else {
402  return fopen(buf, mode);
403  }
404  }
405 #else
406  if (restrict_read) {
407  FILE *fptr;

409 
410  old_mask = umask(0077);
