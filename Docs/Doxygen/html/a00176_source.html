80 {

82 
83  if(!mosq) return NULL;
84 
85  pthread_mutex_lock(&mosq->state_mutex);
86  if(mosq->state == mosq_cs_connect_async){
87  pthread_mutex_unlock(&mosq->state_mutex);
88  mosquitto_reconnect(mosq);
89  }else{
90  pthread_mutex_unlock(&mosq->state_mutex);
91  }
92 
93  if(!mosq->keepalive){
94  /* Sleep for a day if keepalive disabled. */
95  mosquitto_loop_forever(mosq, 1000*86400, 1);
96  }else{
97  /* Sleep for our keepalive value. publish() etc. will wake us up. */
98  mosquitto_loop_forever(mosq, mosq->keepalive*1000, 1);
99  }
100
