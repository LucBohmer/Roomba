127 int _mosquitto_handle_pubrel(struct mosquitto_db *db, struct mosquitto *mosq)
128 {
129  uint16_t mid;
130 #ifndef WITH_BROKER
131  struct mosquitto_message_all *message = NULL;
132 #endif
133  int rc;
134 
135  assert(mosq);
136  if(mosq->protocol == mosq_p_mqtt311){
137  if((mosq->in_packet.command&0x0F) != 0x02){
138  return MOSQ_ERR_PROTOCOL;
139  }
140  }
141  rc = _mosquitto_read_uint16(&mosq->in_packet, &mid);
142  if(rc) return rc;
143 #ifdef WITH_BROKER
144  _mosquitto_log_printf(NULL, MOSQ_LOG_DEBUG, "Received PUBREL from %s (Mid: %d)", mosq->id, mid);
145 
146  if(mqtt3_db_message_release(db, mosq, mid, mosq_md_in)){
147  /* Message not found. Still send a PUBCOMP anyway because this could be
148  * due to a repeated PUBREL after a client has reconnected. */
149  _mosquitto_log_printf(mosq, MOSQ_LOG_WARNING, "Warning: Received PUBREL from %s for an unknown packet identifier %d.", mosq->id, mid);
150  }
151 #else
152  _mosquitto_log_printf(mosq, MOSQ_LOG_DEBUG, "Client %s received PUBREL (Mid: %d)", mosq->id, mid);
153 
154  if(!_mosquitto_message_remove(mosq, mid, mosq_md_in, &message)){
155  /* Only pass the message on if we have removed it from the queue - this
156  * prevents multiple callbacks for the same message. */
157  pthread_mutex_lock(&mosq->callback_mutex);
158  if(mosq->on_message){
159  mosq->in_callback = true;
160  mosq->on_message(mosq, mosq->userdata, &message->msg);
161  mosq->in_callback = false;
162  }
163  pthread_mutex_unlock(&mosq->callback_mutex);
164  _mosquitto_message_cleanup(&message);
165  }
166 #endif
167  rc = _mosquitto_send_pubcomp(mosq, mid);
168  if(rc) return rc;
169 
170  return MOSQ_ERR_SUCCESS;
171 }

173 int _mosquitto_handle_suback(struct mosquitto *mosq)
174 {
175  uint16_t mid;
176  uint8_t qos;
177  int *granted_qos;
178  int qos_count;
179  int i = 0;
180  int rc;
181 
182  assert(mosq);
183 #ifdef WITH_BROKER
184  _mosquitto_log_printf(NULL, MOSQ_LOG_DEBUG, "Received SUBACK from %s", mosq->id);
185 #else
186  _mosquitto_log_printf(mosq, MOSQ_LOG_DEBUG, "Client %s received SUBACK", mosq->id);
187 #endif
188  rc = _mosquitto_read_uint16(&mosq->in_packet, &mid);
189  if(rc) return rc;
190 
191  qos_count = mosq->in_packet.remaining_length - mosq->in_packet.pos;
192  granted_qos = _mosquitto_malloc(qos_count*sizeof(int));
193  if(!granted_qos) return MOSQ_ERR_NOMEM;
194  while(mosq->in_packet.pos < mosq->in_packet.remaining_length){
195  rc = _mosquitto_read_byte(&mosq->in_packet, &qos);
196  if(rc){
197  _mosquitto_free(granted_qos);
198  return rc;
199  }
200  granted_qos[i] = (int)qos;
201  i++;
202  }
203 #ifndef WITH_BROKER
204  pthread_mutex_lock(&mosq->callback_mutex);
205  if(mosq->on_subscribe){
206  mosq->in_callback = true;
207  mosq->on_subscribe(mosq, mosq->userdata, mid, qos_count, granted_qos);
208  mosq->in_callback = false;
209  }
210  pthread_mutex_unlock(&mosq->callback_mutex);
211 #endif
212  _mosquitto_free(granted_qos);
213 
214  return MOSQ_ERR_SUCCESS;
215 }
