159 int load_file(const char *filename)
160 {
161  long pos, rlen;
162  FILE *fptr = NULL;
163 
164  fptr = fopen(filename, "rb");
165  if(!fptr){
166  if(!quiet) fprintf(stderr, "Error: Unable to open file \"%s\".\n", filename);
167  return 1;
168  }
169  mode = MSGMODE_FILE;
170  fseek(fptr, 0, SEEK_END);
171  msglen = ftell(fptr);
172  if(msglen > 268435455){
173  fclose(fptr);
174  if(!quiet) fprintf(stderr, "Error: File \"%s\" is too large (>268,435,455 bytes).\n", filename);
175  return 1;
176  }else if(msglen == 0){
177  fclose(fptr);
178  if(!quiet) fprintf(stderr, "Error: File \"%s\" is empty.\n", filename);
179  return 1;
180  }else if(msglen < 0){
181  fclose(fptr);
182  if(!quiet) fprintf(stderr, "Error: Unable to determine size of file \"%s\".\n", filename);
183  return 1;
184  }
185  fseek(fptr, 0, SEEK_SET);
186  message = malloc(msglen);
187  if(!message){
188  fclose(fptr);
189  if(!quiet) fprintf(stderr, "Error: Out of memory.\n");
190  return 1;
191  }
192  pos = 0;
193  while(pos < msglen){
194  rlen = fread(&(message[pos]), sizeof(char), msglen-pos, fptr);
195  pos += rlen;
196  }
197  fclose(fptr);
198  return 0;
199 }

290 int main(int argc, char *argv[])
291 {
292  struct mosq_config cfg;
293  struct mosquitto *mosq = NULL;
294  int rc;
295  int rc2;
296  char *buf;
297  int buf_len = 1024;
298  int buf_len_actual;
299  int read_len;
300  int pos;
301 
302  buf = malloc(buf_len);
303  if(!buf){
304  fprintf(stderr, "Error: Out of memory.\n");
305  return 1;
306  }
307 
308  memset(&cfg, 0, sizeof(struct mosq_config));
309  rc = client_config_load(&cfg, CLIENT_PUB, argc, argv);
310  if(rc){
311  client_config_cleanup(&cfg);
312  if(rc == 2){
313  /* --help */
314  print_usage();
315  }else{
316  fprintf(stderr, "\nUse 'mosquitto_pub --help' to see usage.\n");
317  }
318  return 1;
319  }
320 
321  topic = cfg.topic;
322  message = cfg.message;
323  msglen = cfg.msglen;
324  qos = cfg.qos;
325  retain = cfg.retain;
326  mode = cfg.pub_mode;
327  username = cfg.username;
328  password = cfg.password;
329  quiet = cfg.quiet;
330 
331  if(cfg.pub_mode == MSGMODE_STDIN_FILE){
332  if(load_stdin()){
333  fprintf(stderr, "Error loading input from stdin.\n");
334  return 1;
335  }
336  }else if(cfg.file_input){
337  if(load_file(cfg.file_input)){
338  fprintf(stderr, "Error loading input file \"%s\".\n", cfg.file_input);
339  return 1;
340  }
341  }
342 
343  if(!topic || mode == MSGMODE_NONE){
344  fprintf(stderr, "Error: Both topic and message must be supplied.\n");
345  print_usage();
346  return 1;
347  }
348 
349 
350  mosquitto_lib_init();
351 
352  if(client_id_generate(&cfg, "mosqpub")){
353  return 1;
354  }
355 
356  mosq = mosquitto_new(cfg.id, true, NULL);
357  if(!mosq){
358  switch(errno){
359  case ENOMEM:
360  if(!quiet) fprintf(stderr, "Error: Out of memory.\n");
361  break;
362  case EINVAL:
363  if(!quiet) fprintf(stderr, "Error: Invalid id.\n");
364  break;
365  }
366  mosquitto_lib_cleanup();
367  return 1;
368  }
369  if(cfg.debug){
370  mosquitto_log_callback_set(mosq, my_log_callback);
371  }
372  mosquitto_connect_callback_set(mosq, my_connect_callback);
373  mosquitto_disconnect_callback_set(mosq, my_disconnect_callback);
374  mosquitto_publish_callback_set(mosq, my_publish_callback);
375 
376  if(client_opts_set(mosq, &cfg)){
377  return 1;
378  }
379  rc = client_connect(mosq, &cfg);
380  if(rc) return rc;
381 
382  if(mode == MSGMODE_STDIN_LINE){
383  mosquitto_loop_start(mosq);
384  }
385 
386  do{
387  if(mode == MSGMODE_STDIN_LINE){
388  if(status == STATUS_CONNACK_RECVD){
389  pos = 0;
390  read_len = buf_len;
391  while(fgets(&buf[pos], read_len, stdin)){
392  buf_len_actual = strlen(buf);
393  if(buf[buf_len_actual-1] == '\n'){
394  buf[buf_len_actual-1] = '\0';
395  rc2 = mosquitto_publish(mosq, &mid_sent, topic, buf_len_actual-1, buf, qos, retain);
396  if(rc2){
397  if(!quiet) fprintf(stderr, "Error: Publish returned %d, disconnecting.\n", rc2);
398  mosquitto_disconnect(mosq);
399  }
400  break;
401  }else{
402  buf_len += 1024;
403  pos += 1023;
404  read_len = 1024;
405  buf = realloc(buf, buf_len);
406  if(!buf){
407  fprintf(stderr, "Error: Out of memory.\n");
408  return 1;
409  }
410  }
411  }
412  if(feof(stdin)){
413  last_mid = mid_sent;
414  status = STATUS_WAITING;
415  }
416  }else if(status == STATUS_WAITING){
417  if(last_mid_sent == last_mid && disconnect_sent == false){
418  mosquitto_disconnect(mosq);
419  disconnect_sent = true;
420  }
421 #ifdef WIN32
422  Sleep(100);
423 #else
424  usleep(100000);
425 #endif
426  }
427  rc = MOSQ_ERR_SUCCESS;
428  }else{
429  rc = mosquitto_loop(mosq, -1, 1);
430  }
431  }while(rc == MOSQ_ERR_SUCCESS && connected);
432 
433  if(mode == MSGMODE_STDIN_LINE){
434  mosquitto_loop_stop(mosq, false);
435  }
436 
437  if(message && mode == MSGMODE_FILE){
438  free(message);
439  }
440  mosquitto_destroy(mosq);
441  mosquitto_lib_cleanup();
442 
443  if(rc){
444  fprintf(stderr, "Error: %s\n", mosquitto_strerror(rc));
445  }
446  return rc;
447 }
